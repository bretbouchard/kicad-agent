       ```python
       """KiCad adapter: bidirectional SchematicIR <-> AbstractCircuit conversion.

       This is the reference adapter that all other format adapters follow.
       It proves that format-specific data can be losslessly extracted into
       Abstract AST and reconstructed.
       """

       from __future__ import annotations

       import logging
       from pathlib import Path
       from typing import Any, Optional

       from kicad_agent.abstract_ast.models import (
           AbstractCircuit,
           AbstractComponent,
           AbstractNet,
           AbstractPin,
           AbstractSheet,
           PinType,
           Position,
           WireSegment,
       )
       from kicad_agent.ir.schematic_ir import SchematicIR

       logger = logging.getLogger(__name__)

       # KiCad electrical type string -> PinType enum
       # Maps kiutils pin electrical type values to format-agnostic PinType
       KICAD_PIN_TYPE_MAP: dict[str, PinType] = {
           "input": PinType.INPUT,
           "output": PinType.OUTPUT,
           "bidirectional": PinType.BIDI,
           "tri_state": PinType.BIDI,  # KiCad tri_state maps to BIDI
           "passive": PinType.PASSIVE,
           "unspecified": PinType.UNSPECIFIED,
           "power_in": PinType.POWER_IN,
           "power_out": PinType.POWER_OUT,
           "open_collector": PinType.OUTPUT,  # Approximation
           "open_emitter": PinType.OUTPUT,    # Approximation
           "no_connect": PinType.NO_CONNECT,
           "free": PinType.UNSPECIFIED,       # Approximation
       }

       # Fields preserved in round-trip (KiCad -> Abstract -> KiCad)
       ROUNDTRIP_PRESERVED = {
           "component_ref": True,
           "component_lib_id": True,
           "component_value": True,
           "component_footprint": True,
           "component_position": True,
           "component_rotation": True,
           "pin_names": True,
           "pin_numbers": True,
           "pin_types": True,
           "net_names": True,
           "net_connectivity": True,  # Which pins are connected
           "hierarchical_sheet_refs": True,
           "hierarchical_labels": True,
       }

       # Fields NOT preserved (lost in round-trip)
       ROUNDTRIP_LOST = {
           "wire_geometry": "Wire exact coordinates may shift due to re-routing",
           "graphical_elements": "Lines, arcs, text boxes not in Abstract AST",
           "annotations": "Notes, text comments dropped",
           "uuid_values": "UUIDs regenerated on reconstruction",
           "zone_fills": "PCB zone fills not in schematic Abstract AST",
           "erc_exclusions": "ERC violation suppressions lost",
       }

       LOSS_ACCOUNTING = {
           "preserved": ROUNDTRIP_PRESERVED,
           "lost": ROUNDTRIP_LOST,
       }


       class KiCadAdapter:
           """Bidirectional converter between SchematicIR and AbstractCircuit.

           Usage:
               # KiCad -> Abstract
               circuit = KiCadAdapter.schematic_ir_to_abstract(schematic_ir)

               # Abstract -> KiCad
               new_ir = KiCadAdapter.abstract_to_schematic_ir(circuit)
           """

           @staticmethod
           def schematic_ir_to_abstract(ir: SchematicIR) -> AbstractCircuit:
               """Extract AbstractCircuit from a SchematicIR.

               Converts:
               - SchematicSymbols -> AbstractComponents (with pins from lib_symbols)
               - Wire connectivity + labels -> AbstractNets (with pin_refs)
               - Hierarchical sheets -> AbstractSheets

               Args:
                   ir: SchematicIR wrapping a parsed KiCad schematic.

               Returns:
                   AbstractCircuit with all extractable circuit semantics.
               """
               sch = ir.schematic
               components = _extract_components(sch)
               nets = _extract_nets(sch)
               sheets = _extract_sheets(sch)

               return AbstractCircuit(
                   name=sch.title or "",
                   components=components,
                   nets=nets,
                   sheets=sheets,
                   metadata={
                       "source_format": "kicad_schematic",
                       "source_path": str(ir.file_path),
                   },
               )

           @staticmethod
           def abstract_to_schematic_ir(
               circuit: AbstractCircuit,
           ) -> Any:
               """Generate SchematicIR from AbstractCircuit.

               Creates a NEW KiCad schematic from the AbstractCircuit data.
               Note: This produces a minimal schematic -- wire routing uses
               auto-placement, not original wire geometry.

               Args:
                   circuit: AbstractCircuit to convert.

               Returns:
                   SchematicIR wrapping the generated KiCad schematic.
               """
               from kiutils.schematic import Schematic
               from kiutils.schematic import SchematicSymbol as KiuSym
               from kiutils.schematic import Wire, Label, Sheet, SheetPin
               from kiutils.misc import Position as KiuPos

               sch = Schematic()
               sch.title = circuit.name

               # 1. Convert AbstractComponents -> kiutils SchematicSymbols
               for idx, comp in enumerate(circuit.components):
                   # Auto-place components in a grid layout (50mil spacing)
                   col = idx % 10
                   row = idx // 10
                   x = 25.4 + col * 50.0  # 50mm spacing
                   y = 25.4 + row * 50.0

                   sym = KiuSym(
                       libId=comp.lib_id,
                       reference=comp.ref,
                       value=comp.value,
                       footprint=comp.footprint or "",
                       position=KiuPos(X=x, Y=y, angle=comp.rotation),
                   )
                   # Set properties
                   for prop in [("Reference", comp.ref), ("Value", comp.value)]:
                       pass  # Properties set via constructor fields above
                   sch.schematicSymbols.append(sym)

               # 2. Convert AbstractNets -> wires + labels
               for net in circuit.nets:
                   # Create a global label at origin for each net
                   label = Label(
                       name=net.name,
                       position=KiuPos(X=0, Y=0),
                       effects="global",
                   )
                   sch.labels.append(label)
                   # Wire segments connecting pin positions are created
                   # by the auto-router (Phase 38). Here we place net
                   # labels which guarantee connectivity in KiCad.

               # 3. Convert AbstractSheets -> hierarchical sheet instances
               for sheet_data in circuit.sheets:
                   hsheet = Sheet(
                       fileName=sheet_data.file_name,
                       sheetName=sheet_data.name,
                       position=KiuPos(X=0, Y=0),
                   )
                   sch.sheets.append(hsheet)

               # 4. Wrap in SchematicIR using the existing parser layer
               from kicad_agent.ir.schematic_ir import SchematicIR
               from kicad_agent.parser.types import ParseResult
               import tempfile

               # Serialize to temp file then re-parse through SchematicIR
               with tempfile.NamedTemporaryFile(
                   suffix=".kicad_sch", mode="w", delete=False
               ) as f:
                   sch.serialize(f.name)
                   from pathlib import Path
                   return SchematicIR(
                       _parse_result=ParseResult(
                           kiutils_obj=sch,
                           raw_content="",
                           file_path=Path(f.name),
                           file_type="kicad_sch",
                       )
                   )


       def _extract_components(sch: Any) -> list[AbstractComponent]:
           """Extract components from kiutils Schematic."""
           components = []
           # Build lib_symbols index for pin extraction
           lib_symbols = _build_lib_symbols_index(sch)

           for sym in sch.schematicSymbols:
               ref = _get_property(sym, "Reference")
               if not ref or ref == "#PWR" or ref.startswith("#"):
                   continue  # Skip power flags and virtual symbols

               lib_id = sym.libId
               value = _get_property(sym, "Value") or ""
               footprint = _get_property(sym, "Footprint") or None

               # Position and rotation
               pos = None
               rotation = 0.0
               if hasattr(sym, 'position') and sym.position:
                   pos = Position(x=sym.position.X, y=sym.position.Y)
                   rotation = sym.position.angle if hasattr(sym.position, 'angle') else 0.0

               # Extract pins from lib_symbols (not stub instances)
               pins = _extract_pins(sym, lib_symbols)

               components.append(AbstractComponent(
                   ref=ref,
                   lib_id=lib_id,
                   value=value,
                   footprint=footprint,
                   position=pos,
                   rotation=rotation,
                   pins=pins,
               ))

           return components


       def _extract_nets(sch: Any) -> list[AbstractNet]:
           """Derive nets from wire connectivity graph + labels.

           Algorithm:
           1. Collect all wire segments with coordinates
           2. Collect all labels with positions and names
           3. Build connectivity graph: wires sharing endpoints are same net
           4. Attach labels to nets by position proximity
           5. Find component pins at wire endpoints -> pin_refs
           """
           # Collect wire segments as WireSegments
           wires: list[WireSegment] = []
           if hasattr(sch, 'wires'):
               for w in sch.wires:
                   wires.append(WireSegment(
                       x1=w.start.X, y1=w.start.Y,
                       x2=w.end.X, y2=w.end.Y,
                   ))

           # Collect all labels (global + local + hierarchical)
           labels_by_pos: dict[tuple[float, float], str] = {}
           if hasattr(sch, 'labels'):
               for lbl in sch.labels:
                   labels_by_pos[(lbl.position.X, lbl.position.Y)] = lbl.name
           if hasattr(sch, 'globalLabels'):
               for lbl in sch.globalLabels:
                   labels_by_pos[(lbl.position.X, lbl.position.Y)] = lbl.name
           if hasattr(sch, 'hierarchicalLabels'):
               for lbl in sch.hierarchicalLabels:
                   labels_by_pos[(lbl.position.X, lbl.position.Y)] = lbl.name

           # Build connectivity graph using Union-Find on wire endpoints
           # Each point is keyed by rounded coordinates (0.01mm precision)
           parent: dict[tuple[float, float], tuple[float, float]] = {}

           def _key(x: float, y: float) -> tuple[float, float]:
               return (round(x, 2), round(y, 2))

           def _find(k: tuple[float, float]) -> tuple[float, float]:
               while parent.get(k, k) != k:
                   parent[k] = parent.get(parent[k], parent[k])
                   k = parent[k]
               return k

           def _union(a: tuple, b: tuple) -> None:
               ra, rb = _find(a), _find(b)
               if ra != rb:
                   parent[ra] = rb

           for ws in wires:
               _union(_key(ws.x1, ws.y1), _key(ws.x2, ws.y2))

           # Group wire endpoints into net clusters
           clusters: dict[tuple, set[tuple]] = {}
           for ws in wires:
               root = _find(_key(ws.x1, ws.y1))
               clusters.setdefault(root, set())
               clusters[root].add(_key(ws.x1, ws.y1))
               clusters[root].add(_key(ws.x2, ws.y2))

           # Build pin position index for pin_ref matching
           pin_positions: dict[tuple[float, float], tuple[str, str]] = {}
           if hasattr(sch, 'schematicSymbols'):
               for sym in sch.schematicSymbols:
                   ref = _get_property(sym, "Reference")
                   if not ref or ref.startswith("#"):
                       continue
                   pos = sym.position if hasattr(sym, 'position') else None
                   if pos:
                       # Pin positions derived from lib_symbols
                       lib_sym = _build_lib_symbols_index(sch).get(sym.libId)
                       if lib_sym and hasattr(lib_sym, 'pins'):
                           for pin in lib_sym.pins:
                               px = pos.X + getattr(pin, 'posX', 0)
                               py = pos.Y + getattr(pin, 'posY', 0)
                               pin_num = getattr(pin, 'number', '?')
                               pin_positions[_key(px, py)] = (ref, pin_num)

           # Convert clusters to AbstractNets with names and pin_refs
           nets: list[AbstractNet] = []
           for _root, points in clusters.items():
               # Find net name from labels at any cluster point
               net_name = ""
               pin_refs: list[tuple[str, str]] = []
               for pt in points:
                   if pt in labels_by_pos:
                       net_name = labels_by_pos[pt]
                   if pt in pin_positions:
                       pin_refs.append(pin_positions[pt])
               if net_name or pin_refs:
                   nets.append(AbstractNet(
                       name=net_name or f"Net_{len(nets)}",
                       pin_refs=pin_refs,
                   ))

           return nets


       def _extract_sheets(sch: Any) -> list[AbstractSheet]:
           """Extract hierarchical sheets."""
           sheets: list[AbstractSheet] = []
           if hasattr(sch, 'sheets'):
               for hs in sch.sheets:
                   sheet_name = getattr(hs, 'sheetName', '') or ''
                   file_name = getattr(hs, 'fileName', '') or ''
                   # Extract hierarchical labels (sheet pins)
                   h_labels: list[str] = []
                   if hasattr(hs, 'sheetPins'):
                       for sp in hs.sheetPins:
                           h_labels.append(getattr(sp, 'name', ''))
                   pos = None
                   if hasattr(hs, 'position') and hs.position:
                       pos = Position(x=hs.position.X, y=hs.position.Y)
                   sheets.append(AbstractSheet(
                       name=sheet_name,
                       file_name=file_name,
                       labels=h_labels,
                       position=pos,
                   ))
           return sheets


       def _build_lib_symbols_index(sch: Any) -> dict[str, Any]:
           """Build lib_id -> symbol definition index for pin extraction."""
           index = {}
           if hasattr(sch, 'libSymbols') and sch.libSymbols:
               for lib_sym in sch.libSymbols:
                   index[lib_sym.libId] = lib_sym
           return index


       def _extract_pins(sym: Any, lib_symbols: dict) -> list[AbstractPin]:
           """Extract pins from lib_symbols definition for a component."""
           pins: list[AbstractPin] = []
           lib_id = getattr(sym, 'libId', '')
           lib_sym = lib_symbols.get(lib_id)
           if not lib_sym:
               return pins
           if not hasattr(lib_sym, 'pins'):
               return pins
           for pin in lib_sym.pins:
               pin_name = getattr(pin, 'name', '') or ''
               pin_number = str(getattr(pin, 'number', '') or '')
               elec_type = getattr(pin, 'electricalType', 'unspecified') or 'unspecified'
               pin_type = KICAD_PIN_TYPE_MAP.get(elec_type, PinType.UNSPECIFIED)
               pin_pos = None
               if hasattr(pin, 'position') and pin.position:
                   pin_pos = Position(x=pin.position.X, y=pin.position.Y)
               pins.append(AbstractPin(
                   name=pin_name,
                   number=pin_number,
                   pin_type=pin_type,
                   position=pin_pos,
               ))
           return pins


       def _get_property(sym: Any, key: str) -> Optional[str]:
           """Get a property value from a kiutils SchematicSymbol."""
           if hasattr(sym, 'properties'):
               for prop in sym.properties:
                   if prop.key == key:
                       return prop.value
           return None
       ```
