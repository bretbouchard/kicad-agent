       ```python
       """Visual diff engine for comparing two KiCad schematic SVGs.

       Produces a diff SVG with green highlights for added elements,
       red highlights for removed elements, and a color legend.
       """

       from __future__ import annotations

       import logging
       import xml.etree.ElementTree as ET
       from pathlib import Path

       from pydantic import BaseModel, Field

       from kicad_agent.spatial.svg_utils import (
           SVG_NS,
           create_svg_group,
           create_svg_text,
           get_svg_dimensions,
           parse_svg,
           write_svg,
       )

       logger = logging.getLogger(__name__)


       class VisualDiffResult(BaseModel):
           """Result of a visual diff comparison.

           Attributes:
               added_count: Number of elements in 'after' but not in 'before'.
               removed_count: Number of elements in 'after' but not in 'before' (wait, reversed).
               modified_count: Number of elements that changed between versions.
               diff_svg_path: Path to the generated diff SVG.
               identical: Whether the two SVGs are identical.
           """

           added_count: int = Field(ge=0)
           removed_count: int = Field(ge=0)
           modified_count: int = Field(ge=0)
           diff_svg_path: str
           identical: bool = False


       class VisualDiffer:
           """Compare two KiCad schematic SVGs and highlight differences.

           Diffing strategy: Compare serialized element representations.
           - Added: Elements in 'after' not in 'before' (green highlight)
           - Removed: Elements in 'before' not in 'after' (red highlight)
           - Modified: Same tag+position, different attributes (yellow highlight)
           """

           # Highlight colors
           COLOR_ADDED = "#00cc00"
           COLOR_REMOVED = "#cc0000"
           COLOR_MODIFIED = "#ccaa00"

           def compare(
               self,
               svg_before: Path,
               svg_after: Path,
               output_path: Path,
           ) -> VisualDiffResult:
               """Compare two SVGs and produce a diff SVG.

               Args:
                   svg_before: Path to the 'before' SVG.
                   svg_after: Path to the 'after' SVG.
                   output_path: Path for the diff SVG output.

               Returns:
                   VisualDiffResult with counts and diff SVG path.
               """
               root_before = parse_svg(svg_before)
               root_after = parse_svg(svg_after)

               # Collect element signatures from both SVGs
               sigs_before = self._collect_signatures(root_before)
               sigs_after = self._collect_signatures(root_after)

               set_before = set(sigs_before.keys())
               set_after = set(sigs_after.keys())

               added = set_after - set_before
               removed = set_before - set_after
               # Modified: same position, different content
               common = set_before & set_after
               modified = {
                   sig for sig in common
                   if sigs_before[sig] != sigs_after[sig]
               }

               identical = len(added) == 0 and len(removed) == 0 and len(modified) == 0

               # Build diff SVG (use 'after' as base)
               diff_root = parse_svg(svg_after)
               _, height = get_svg_dimensions(diff_root)

               # Highlight added elements (green overlay)
               for sig in added:
                   elem = sigs_after[sig]
                   self._highlight_element(elem, self.COLOR_ADDED, opacity=0.2)

               # Build overlay for removed elements (we can't highlight what's not there,
               # so we add red markers at their positions)
               if removed:
                   removed_group = create_svg_group(label="removed-elements")
                   for sig in removed:
                       elem = sigs_before[sig]
                       x = elem.attrib.get("x", "0")
                       y = elem.attrib.get("y", "0")
                       marker = ET.Element(
                           f"{{{SVG_NS}}}rect",
                           attrib={
                               "x": str(float(x) - 1), "y": str(float(y) - 1),
                               "width": "5", "height": "3",
                               "fill": self.COLOR_REMOVED, "opacity": "0.3",
                               "stroke": self.COLOR_REMOVED, "stroke-width": "0.3",
                           },
                       )
                       removed_group.append(marker)
                   diff_root.append(removed_group)

               # Add legend
               if not identical:
                   self._add_legend(diff_root, len(added), len(removed), len(modified), height)

               write_svg(diff_root, output_path)

               return VisualDiffResult(
                   added_count=len(added),
                   removed_count=len(removed),
                   modified_count=len(modified),
                   diff_svg_path=str(output_path),
                   identical=identical,
               )

           def _collect_signatures(self, root: ET.Element) -> dict[str, ET.Element]:
               """Collect element signatures for comparison.

               Uses a tuple of (tag, position attributes) as signature key
               and the full serialized element as value.
               """
               signatures = {}
               for elem in root.iter():
                   tag = elem.tag.replace(f"{{{SVG_NS}}}", "")
                   if tag in ("svg", "defs", "style", "metadata"):
                       continue
                   # Build position-based signature
                   x = elem.attrib.get("x", elem.attrib.get("cx", ""))
                   y = elem.attrib.get("y", elem.attrib.get("cy", ""))
                   sig = f"{tag}:{x},{y}"
                   if sig not in signatures:
                       signatures[sig] = elem
               return signatures

           def _highlight_element(self, elem: ET.Element, color: str, opacity: float = 0.2) -> None:
               """Add a highlight overlay to an element."""
               existing_style = elem.attrib.get("style", "")
               highlight = f"fill:{color};opacity:{opacity}"
               if existing_style:
                   elem.attrib["style"] = existing_style + ";" + highlight
               else:
                   elem.attrib["fill"] = color
                   elem.attrib["opacity"] = str(opacity)

           def _add_legend(
               self, root: ET.Element, added: int, removed: int, modified: int, svg_height: float,
           ) -> None:
               """Add diff color legend to SVG."""
               legend = create_svg_group(label="diff-legend")
               y_start = svg_height + 5

               entries = [
                   (f"+ Added: {added}", self.COLOR_ADDED),
                   (f"- Removed: {removed}", self.COLOR_REMOVED),
                   (f"~ Modified: {modified}", self.COLOR_MODIFIED),
               ]
               for i, (text, color) in enumerate(entries):
                   t = create_svg_text(
                       x=2, y=y_start + i * 3 + 2,
                       text=text, font_size=2.5, fill=color,
                   )
                   legend.append(t)

               root.append(legend)
       ```

       ```python
       """Markdown report generator for demo pipeline output.

       Produces rich Markdown reports with embedded SVG references,
       ERC statistics, before/after comparisons, and timing information.
       """

       from __future__ import annotations

       import logging
       from pathlib import Path

       from kicad_agent.demo.pipeline import DemoReport

       logger = logging.getLogger(__name__)


       class ReportGenerator:
           """Generate Markdown reports from DemoReport data.

           Args:
               include_svg_refs: Whether to include SVG image references.
           """

           def __init__(self, include_svg_refs: bool = True) -> None:
               self.include_svg_refs = include_svg_refs

           def generate(self, report: DemoReport) -> str:
               """Generate a Markdown report from a DemoReport.

               Args:
                   report: Demo pipeline result.

               Returns:
                   Markdown string.
               """
               lines: list[str] = []

               # Header
               lines.append(f"# Demo Report: {report.template_used}")
               lines.append("")
               status = "SUCCESS" if report.success else "FAILED"
               lines.append(f"**Status:** {status} | **Duration:** {report.duration_seconds:.1f}s")
               lines.append("")

               # Summary
               lines.append("## Summary")
               lines.append("")
               lines.append(f"| Field | Value |")
               lines.append(f"|-------|-------|")
               lines.append(f"| Template | `{report.template_used}` |")
               lines.append(f"| Duration | {report.duration_seconds:.1f}s |")
               lines.append(f"| Stages | {', '.join(report.stages_completed) or 'None'} |")
               lines.append(f"| Project | `{report.project_dir or 'N/A'}` |")
               lines.append("")

               # ERC Results
               lines.append("## ERC Results")
               lines.append("")
               if report.erc_before is not None:
                   lines.append(f"| Metric | Count |")
                   lines.append(f"|--------|-------|")
                   lines.append(f"| Violations (before fix) | {report.erc_before} |")
                   lines.append(f"| Violations (after fix) | {report.erc_after or 'N/A'} |")
                   if report.erc_after is not None and report.erc_before > 0:
                       delta = report.erc_after - report.erc_before
                       fixed = abs(delta) if delta < 0 else 0
                       lines.append(f"| Auto-fixed | {fixed} |")
                   lines.append("")
               else:
                   lines.append("ERC was not run (kicad-cli unavailable or schematic not generated).")
                   lines.append("")

               # SVG Output
               if report.svg_paths:
                   lines.append("## Generated Files")
                   lines.append("")
                   for svg_path in report.svg_paths:
                       lines.append(f"- `{svg_path}`")
                       if self.include_svg_refs:
                           lines.append("")
                           lines.append(f"![{report.template_used}]({svg_path})")
                           lines.append("")
                   lines.append("")

               # Errors
               if report.errors:
                   lines.append("## Errors")
                   lines.append("")
                   for error in report.errors:
                       lines.append(f"- {error}")
                   lines.append("")

               return "\n".join(lines)

           def generate_and_write(self, report: DemoReport, output_path: Path) -> Path:
               """Generate report and write to file.

               Args:
                   report: Demo pipeline result.
                   output_path: Path for the Markdown output file.

               Returns:
                   Path to the written file.
               """
               content = self.generate(report)
               output_path.parent.mkdir(parents=True, exist_ok=True)
               output_path.write_text(content, encoding="utf-8")
               return output_path
       ```
