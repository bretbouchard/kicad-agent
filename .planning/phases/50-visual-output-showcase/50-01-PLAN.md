       """SVG parsing and manipulation utilities for KiCad-generated schematics.

       Handles KiCad SVG namespace quirks, coordinate transforms, and
       element creation for annotation overlays.
       """

       from __future__ import annotations

       import xml.etree.ElementTree as ET
       from pathlib import Path
       from typing import Any


       # KiCad SVG namespace constants
       SVG_NS = "http://www.w3.org/2000/svg"
       XLINK_NS = "http://www.w3.org/1999/xlink"

       # Default mm-per-pixel for KiCad SVG (KiCad uses mm in viewBox)
       DEFAULT_SCALE = 1.0


       def parse_svg(path: Path) -> ET.Element:
           """Parse an SVG file and return the root element.

           Args:
               path: Path to SVG file.

           Returns:
               Root Element of the parsed SVG.

           Raises:
               FileNotFoundError: If path does not exist.
               ValueError: If file is not valid SVG.
           """
           if not path.exists():
               raise FileNotFoundError(f"SVG file not found: {path}")

           try:
               tree = ET.parse(str(path))
               root = tree.getroot()
           except ET.ParseError as exc:
               raise ValueError(f"Invalid SVG: {exc}") from exc

           # Validate it looks like SVG (namespace or tag check)
           tag = root.tag
           if not (tag.endswith("svg") or "svg" in tag):
               raise ValueError(f"Not an SVG file (root tag: {tag})")

           return root


       def get_svg_dimensions(root: ET.Element) -> tuple[float, float]:
           """Extract dimensions in mm from SVG viewBox.

           Args:
               root: SVG root element.

           Returns:
               (width_mm, height_mm) tuple.
           """
           viewbox = root.attrib.get("viewBox", root.attrib.get("viewbox", ""))
           if viewbox:
               parts = viewbox.split()
               if len(parts) == 4:
                   return float(parts[2]), float(parts[3])

           # Fallback: width/height attributes
           w = root.attrib.get("width", "297mm").replace("mm", "")
           h = root.attrib.get("height", "210mm").replace("mm", "")
           return float(w), float(h)


       def svg_to_mm(root: ET.Element, svg_x: float, svg_y: float) -> tuple[float, float]:
           """Convert SVG coordinate to mm using viewBox transform.

           KiCad SVGs use mm in the viewBox, so coordinates are typically 1:1.
           This handles the general case where viewBox offset is non-zero.

           Args:
               root: SVG root element (for viewBox extraction).
               svg_x: X coordinate in SVG units.
               svg_y: Y coordinate in SVG units.

           Returns:
               (x_mm, y_mm) tuple.
           """
           viewbox = root.attrib.get("viewBox", root.attrib.get("viewbox", ""))
           if viewbox:
               parts = viewbox.split()
               if len(parts) == 4:
                   offset_x = float(parts[0])
                   offset_y = float(parts[1])
                   return svg_x - offset_x, svg_y - offset_y
           return svg_x, svg_y


       def create_svg_circle(
           cx: float, cy: float, r: float,
           fill: str = "red", opacity: float = 0.3, stroke: str = "red", stroke_width: float = 0.5,
       ) -> ET.Element:
           """Create an SVG circle element for annotation markers.

           Args:
               cx: Center X in SVG units.
               cy: Center Y in SVG units.
               r: Radius in SVG units.
               fill: Fill color.
               opacity: Fill opacity (0-1).
               stroke: Stroke color.
               stroke_width: Stroke width.

           Returns:
               SVG circle Element.
           """
           return ET.Element(
               f"{{{SVG_NS}}}circle",
               attrib={
                   "cx": str(cx), "cy": str(cy), "r": str(r),
                   "fill": fill, "opacity": str(opacity),
                   "stroke": stroke, "stroke-width": str(stroke_width),
               },
           )


       def create_svg_text(
           x: float, y: float, text: str,
           font_size: float = 2.5, fill: str = "red", font_weight: str = "bold",
       ) -> ET.Element:
           """Create an SVG text element for annotation labels.

           Args:
               x: X position in SVG units.
               y: Y position in SVG units.
               text: Text content.
               font_size: Font size in SVG units.
               fill: Text color.
               font_weight: Font weight.

           Returns:
               SVG text Element.
           """
           elem = ET.Element(
               f"{{{SVG_NS}}}text",
               attrib={
                   "x": str(x), "y": str(y),
                   "font-size": str(font_size), "fill": fill,
                   "font-weight": font_weight,
                   "font-family": "monospace",
               },
           )
           elem.text = text
           return elem


       def create_svg_group(
           label: str = "annotations", transform: str | None = None,
       ) -> ET.Element:
           """Create an SVG group element for containing annotations.

           Args:
               label: Group label (inkscape:label or id).
               transform: Optional SVG transform attribute.

           Returns:
               SVG g Element.
           """
           attrib: dict[str, str] = {"id": label}
           if transform:
               attrib["transform"] = transform
           return ET.Element(f"{{{SVG_NS}}}g", attrib=attrib)


       def write_svg(root: ET.Element, path: Path) -> None:
           """Write SVG element tree to file.

           Args:
               root: SVG root element.
               path: Output file path.
           """
           tree = ET.ElementTree(root)
           ET.indent(tree, space="  ")
           tree.write(str(path), xml_declaration=True, encoding="UTF-8")
       ```

       """SVG annotator: overlays ERC violation markers on schematic SVGs.

       Takes a KiCad schematic SVG and a list of violations, produces
       an annotated SVG with numbered red circles at violation positions,
       callout labels, and a summary legend at the bottom.
       """

       from __future__ import annotations

       import logging
       import xml.etree.ElementTree as ET
       from pathlib import Path
       from typing import Sequence

       from pydantic import BaseModel, Field, field_validator

       from kicad_agent.ops.erc_parser import ErcViolation
       from kicad_agent.spatial.svg_utils import (
           SVG_NS,
           create_svg_circle,
           create_svg_group,
           create_svg_text,
           get_svg_dimensions,
           parse_svg,
           write_svg,
       )

       logger = logging.getLogger(__name__)


       class AnnotationStyle(BaseModel):
           """Style configuration for SVG annotations.

           Attributes:
               color: Marker color (CSS color name or hex).
               font_size: Callout text font size in mm.
               circle_radius: Marker circle radius in mm.
               opacity: Marker fill opacity (0=transparent, 1=opaque).
               legend_font_size: Legend text font size in mm.
               legend_padding: Padding between legend entries in mm.
           """

           color: str = "red"
           font_size: float = Field(default=2.5, gt=0)
           circle_radius: float = Field(default=2.0, gt=0)
           opacity: float = Field(default=0.3, ge=0, le=1)
           legend_font_size: float = Field(default=2.0, gt=0)
           legend_padding: float = Field(default=3.0, gt=0)


       class SvgAnnotator:
           """Annotate SVG schematics with violation markers.

           Args:
               style: Annotation style configuration.
           """

           def __init__(self, style: AnnotationStyle | None = None) -> None:
               self.style = style or AnnotationStyle()

           def annotate(
               self,
               svg_path: Path,
               violations: Sequence[ErcViolation],
               output_path: Path | None = None,
           ) -> Path:
               """Annotate SVG with violation markers.

               Args:
                   svg_path: Path to input SVG schematic.
                   violations: List of ERC violations with positions.
                   output_path: Output path (defaults to input with -annotated suffix).

               Returns:
                   Path to annotated SVG file.
               """
               if output_path is None:
                   output_path = svg_path.with_name(
                       f"{svg_path.stem}-annotated{svg_path.suffix}"
                   )

               root = parse_svg(svg_path)
               width_mm, height_mm = get_svg_dimensions(root)

               if not violations:
                   write_svg(root, output_path)
                   return output_path

               # Create annotation overlay group
               overlay = create_svg_group(label="erc-annotations")
               legend = create_svg_group(label="erc-legend")

               numbered = []
               for idx, violation in enumerate(violations, start=1):
                   if not violation.positions:
                       logger.debug("Skipping violation without position: %s", violation.description)
                       continue
                   for pos_idx, (x, y) in enumerate(violation.positions):
                       marker_id = f"{idx}" if len(violation.positions) == 1 else f"{idx}.{pos_idx+1}"
                       self._add_marker(overlay, x, y, marker_id)
                       numbered.append((marker_id, violation))

               # Build legend
               if numbered:
                   self._build_legend(legend, numbered, height_mm)

               root.append(overlay)
               root.append(legend)

               write_svg(root, output_path)
               return output_path

           def _add_marker(
               self, group: ET.Element, x: float, y: float, label: str,
           ) -> None:
               """Add a numbered circle marker at the given position."""
               circle = create_svg_circle(
                   cx=x, cy=y, r=self.style.circle_radius,
                   fill=self.style.color, opacity=self.style.opacity,
               )
               text = create_svg_text(
                   x=x + self.style.circle_radius + 0.5,
                   y=y + self.style.font_size / 3,
                   text=f"{label}",
                   font_size=self.style.font_size,
                   fill=self.style.color,
               )
               group.append(circle)
               group.append(text)

           def _build_legend(
               self,
               legend: ET.Element,
               entries: list[tuple[str, ErcViolation]],
               svg_height: float,
           ) -> None:
               """Build legend summary at bottom of SVG."""
               y_start = svg_height + 5  # Below the schematic

               # Legend background
               bg = ET.Element(
                   f"{{{SVG_NS}}}rect",
                   attrib={
                       "x": "0", "y": str(y_start - 2),
                       "width": "200", "height": str(len(entries) * self.style.legend_padding + 4),
                       "fill": "white", "stroke": self.style.color, "stroke-width": "0.5",
                   },
               )
               legend.append(bg)

               # Title
               title = create_svg_text(
                   x=2, y=y_start + self.style.legend_font_size,
                   text="ERC Violations:",
                   font_size=self.style.legend_font_size + 0.5,
                   fill=self.style.color,
               )
               legend.append(title)

               for i, (marker_id, violation) in enumerate(entries):
                   y = y_start + (i + 1) * self.style.legend_padding + self.style.legend_font_size
                   line = create_svg_text(
                       x=2, y=y,
                       text=f"[{marker_id}] {violation.severity}: {violation.description}",
                       font_size=self.style.legend_font_size,
                       fill="black" if violation.severity == "warning" else self.style.color,
                   )
                   legend.append(line)
       ```
