Altium SchDoc binary structure (community-documented):
```
.SchDoc (OLE Compound Document)
  ├── FileHeader stream
  │   ├── Record headers: (record_type, size) pairs
  │   └── Record bodies: binary parameter blocks
  ├── Additional streams (optional)
  └── Storage directories

Record types (hex codes):
  0x01 = Component
  0x02 = Pin
  0x03 = IEEE_SYMBOL
  0x04 = Label
  0x05 = Bezier
  0x06 = Polyline
  0x07 = Polygon
  0x08 = Rectangle
  0x09 = RoundRectangle
  0x0A = EllipticalArc
  0x0B = Ellipse
  0x0C = Pie
  0x0D = Line
  0x0E = Arc
  0x0F = Text
  0x10 = Note
  0x11 = CompileMask
  0x12 = SheetSymbol
  0x13 = SheetEntry
  0x14 = ParameterSet
  0x15 = Designator
  0x22 = NetLabel
  0x23 = Wire
  0x27 = Junction
  0x29 = PowerObject

Key parameter blocks (S-expression like):
  |RECORD=1|DESIGNATOR=U1|LIBREFERENCE=NE5532|PARTDESCRIPTION=Dual Op-Amp|
  |RECORD=2|OWNERINDEX=0|NAME=OUT+|PINLENGTH=20|ELECTRICAL=2|
  |RECORD=34|OWNERINDEX=0|NET=VCC|X=100|Y=200|
  |RECORD=1|OWNERPARTID=1|LOCATION.X=500|LOCATION.Y=300|
```

       ```python
       """Altium electrical type code to PinType mapping.

       Altium uses numeric codes for pin electrical types (0-9).
       This mapper converts them to the format-agnostic PinType enum.
       """

       from __future__ import annotations

       from kicad_agent.abstract_ast.models import PinType

       # Altium electrical type codes (from Altium SDK documentation + community)
       # 0 = Passive (same as 4 in some versions)
       # 1 = Input
       # 2 = Output
       # 3 = Bidirectional / IO
       # 4 = Passive
       # 5 = Unspecified / Any
       # 6 = Power In (supply)
       # 7 = Power Out (source)
       # 8 = Open Collector
       # 9 = Open Eitter
       # 10 = No Connect (some versions)
       _ALTIUM_TO_PINTYPE: dict[int, PinType] = {
           0: PinType.PASSIVE,        # Passive (alt code)
           1: PinType.INPUT,          # Input
           2: PinType.OUTPUT,         # Output
           3: PinType.BIDI,           # Bidirectional / IO
           4: PinType.PASSIVE,        # Passive
           5: PinType.UNSPECIFIED,    # Unspecified / Any
           6: PinType.POWER_IN,       # Power In (supply)
           7: PinType.POWER_OUT,      # Power Out (source)
           8: PinType.OUTPUT,         # Open Collector -> approximated as OUTPUT
           9: PinType.OUTPUT,         # Open Emitter -> approximated as OUTPUT
           10: PinType.NO_CONNECT,    # No Connect
       }


       class AltiumSymbolMapper:
           """Map Altium electrical type codes to format-agnostic PinType.

           Usage:
               mapper = AltiumSymbolMapper()
               pin_type = mapper.map_electrical_type(1)  # -> PinType.INPUT
           """

           def map_electrical_type(self, code: int) -> PinType:
               """Map Altium electrical type code to PinType.

               Args:
                   code: Altium numeric electrical type code (0-10).

               Returns:
                   PinType enum value. Returns PinType.UNSPECIFIED for unknown codes.
               """
               return _ALTIUM_TO_PINTYPE.get(code, PinType.UNSPECIFIED)

           def known_codes(self) -> set[int]:
               """Return set of known Altium electrical type codes."""
               return set(_ALTIUM_TO_PINTYPE.keys())
       ```

       ```python
       """Altium .SchDoc parser: binary OLE -> AbstractCircuit (read-only).

       Altium SchDoc files are OLE compound documents containing a FileHeader
       stream with binary records. Each record has a type code and parameter
       block with pipe-delimited key=value pairs.

       This parser is READ-ONLY -- it extracts circuit data into AbstractCircuit
       but cannot write .SchDoc files. For KiCad migration, combine with the
       KiCadAdapter to write the output.
       """

       from __future__ import annotations

       import logging
       import struct
       from pathlib import Path
       from typing import Any, Optional

       from kicad_agent.abstract_ast.models import (
           AbstractCircuit,
           AbstractComponent,
           AbstractNet,
           AbstractPin,
           PinType,
           Position,
           WireSegment,
       )
       from kicad_agent.formats.altium_symbol_mapper import AltiumSymbolMapper
       from kicad_agent.formats.altium_types import (
           AltiumComponent,
           AltiumNetLabel,
           AltiumPin,
           AltiumPowerObject,
           AltiumSheetSymbol,
           AltiumWire,
       )

       logger = logging.getLogger(__name__)

       # Altium record type codes
       RECORD_COMPONENT = 1
       RECORD_PIN = 2
       RECORD_WIRE = 34      # 0x22 -> but some refs say 0x23=35
       RECORD_NETLABEL = 34
       RECORD_POWER = 41     # 0x29
       RECORD_SHEET_SYMBOL = 18  # 0x12

       # Some references use different codes; handle both
       RECORD_WIRE_ALT = 35
       RECORD_NETLABEL_ALT = 35


       class AltiumParser:
           """Parse Altium .SchDoc files into AbstractCircuit (read-only).

           Uses olefile to open the OLE compound document, reads the
           FileHeader stream, and extracts component/pin/wire/net records.

           Usage:
               parser = AltiumParser()
               circuit = parser.parse("design.SchDoc")
               # circuit is now an AbstractCircuit that can be converted
               # to KiCad via KiCadAdapter.abstract_to_schematic_ir()
           """

           def __init__(self) -> None:
               self._symbol_mapper = AltiumSymbolMapper()

           def parse(self, path: str | Path) -> AbstractCircuit:
               """Parse an Altium .SchDoc file into AbstractCircuit.

               Args:
                   path: Path to .SchDoc file.

               Returns:
                   AbstractCircuit with components, pins, and nets.

               Raises:
                   FileNotFoundError: If file does not exist.
                   ValueError: If file is not an OLE compound document.
               """
               import olefile

               resolved = Path(path).resolve()
               if not resolved.exists():
                   raise FileNotFoundError(f"File not found: {resolved}")

               if not olefile.isOleFile(str(resolved)):
                   raise ValueError(
                       f"Not an OLE compound document: {resolved}. "
                       f"SchDoc files must be OLE format."
                   )

               ole = olefile.OleFileIO(str(resolved))

               try:
                   records = self._read_records(ole)
                   components = self._convert_components(records)
                   nets = self._derive_nets(records)
                   sheets = self._convert_sheets(records)

                   return AbstractCircuit(
                       name=resolved.stem,
                       components=components,
                       nets=nets,
                       sheets=sheets,
                       metadata={
                           "source_format": "altium_schdoc",
                           "source_path": str(resolved),
                       },
                   )
               finally:
                   ole.close()

           def _read_records(self, ole: Any) -> dict[str, list]:
               """Read and parse records from FileHeader stream.

               Returns dict mapping record type -> list of parsed records.
               """
               records: dict[str, list] = {
                   "components": [],
                   "pins": [],
                   "wires": [],
                   "net_labels": [],
                   "power_objects": [],
                   "sheet_symbols": [],
               }

               if not ole.exists("FileHeader"):
                   logger.warning("No FileHeader stream found in .SchDoc")
                   return records

               stream = ole.openstream("FileHeader")
               data = stream.read()

               # Parse binary record structure
               # Altium records: 4-byte header (record_type + size) followed by parameters
               # Parameters are pipe-delimited key=value strings
               pos = 0
               while pos < len(data):
                   try:
                       # Record header: record_type (1 byte) + padding (3 bytes) + size (4 bytes)
                       # Simplified parsing -- real Altium format is more complex
                       if pos + 8 > len(data):
                           break

                       record_type = struct.unpack_from("<I", data, pos)[0]
                       record_size = struct.unpack_from("<I", data, pos + 4)[0]
                       pos += 8

                       if pos + record_size > len(data):
                           break

                       record_data = data[pos:pos + record_size]
                       pos += record_size

                       params = self._parse_params(record_data)
                       self._classify_record(record_type, params, records)

                   except (struct.error, UnicodeDecodeError) as e:
                       logger.warning(f"Failed to parse record at offset {pos}: {e}")
                       break

               return records

           def _parse_params(self, data: bytes) -> dict[str, str]:
               """Parse pipe-delimited key=value parameters from record data."""
               params = {}
               try:
                   text = data.decode("ascii", errors="replace")
                   for pair in text.split("|"):
                       if "=" in pair:
                           key, _, value = pair.partition("=")
                           params[key.strip()] = value.strip()
               except Exception:
                   pass
               return params

           def _classify_record(
               self, record_type: int, params: dict[str, str], records: dict
           ) -> None:
               """Classify a parsed record by type and add to records dict."""
               record_num = int(params.get("RECORD", record_type))

               if record_num == RECORD_COMPONENT:
                   records["components"].append(params)
               elif record_num == RECORD_PIN:
                   records["pins"].append(params)
               elif record_num in (RECORD_WIRE, RECORD_WIRE_ALT):
                   records["wires"].append(params)
               elif record_num == RECORD_NETLABEL:
                   records["net_labels"].append(params)
               elif record_num == RECORD_POWER:
                   records["power_objects"].append(params)
               elif record_num == RECORD_SHEET_SYMBOL:
                   records["sheet_symbols"].append(params)

           def _convert_components(self, records: dict) -> list[AbstractComponent]:
               """Convert Altium component records to AbstractComponent."""
               components = []
               comp_params_list = records.get("components", [])
               pin_params_list = records.get("pins", [])

               for comp_params in comp_params_list:
                   ref = comp_params.get("DESIGNATOR", comp_params.get("DESIGNATORDISPLAY", "?"))
                   lib_ref = comp_params.get("LIBREFERENCE", "")
                   description = comp_params.get("PARTDESCRIPTION", "")
                   x = float(comp_params.get("LOCATION.X", "0"))
                   y = float(comp_params.get("LOCATION.Y", "0"))

                   # Find pins belonging to this component (matched by OWNERINDEX)
                   component_index = int(comp_params.get("INDEXINSHEET", "0"))
                   pins = self._extract_component_pins(pin_params_list, component_index)

                   components.append(AbstractComponent(
                       ref=ref,
                       lib_id=lib_ref,
                       value=description,
                       position=Position(x=x, y=y),
                       pins=pins,
                       properties={
                           "altium_lib_reference": lib_ref,
                           "altium_description": description,
                       },
                   ))

               return components

           def _extract_component_pins(
               self, pin_params_list: list[dict[str, str]], owner_index: int
           ) -> list[AbstractPin]:
               """Extract pins belonging to a specific component."""
               pins = []
               for pin_params in pin_params_list:
                   pin_owner = int(pin_params.get("OWNERINDEX", "-1"))
                   if pin_owner == owner_index:
                       name = pin_params.get("NAME", "")
                       number = pin_params.get("NUMBER", pin_params.get("PINNUM", ""))
                       elec_type = int(pin_params.get("ELECTRICAL", "5"))
                       pin_type = self._symbol_mapper.map_electrical_type(elec_type)
                       pins.append(AbstractPin(
                           number=number or name,
                           name=name or number,
                           pin_type=pin_type,
                       ))
               return pins

           def _derive_nets(self, records: dict) -> list[AbstractNet]:
               """Derive nets from wire records + net label records.

               Algorithm:
               1. Collect all wire coordinate pairs
               2. Collect all net labels with positions
       ```
