SpiceLib RawRead API (verified spicelib 1.5.1):
```python
from spicelib import RawRead
raw = RawRead("simulation.raw")
raw.get_trace_names()   # -> list[str] ['time', 'V(vcc)', 'I(R1)', ...]
raw.get_trace("V(vcc)") # -> RawReadDataTrace object
raw.get_wave("V(vcc)")  # -> numpy.ndarray of float values
raw.get_time_axis()     # -> numpy.ndarray (for transient analysis)
raw.get_axis(var)       # -> numpy.ndarray (general axis)
raw.nVariables          # -> int
raw.nPoints             # -> int
raw.raw_type            # -> str (e.g. "transient")
raw.to_dataframe()      # -> pandas DataFrame
raw.get_steps()         # -> step info for stepped simulations
raw.plots               # -> list of plot info
raw.flags               # -> str (raw file flags)
```
