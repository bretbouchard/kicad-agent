import SwiftAISDK
import OpenAIProvider
import AnthropicProvider
import GoogleGenerativeAIProvider

struct AnthropicCloudProvider: LLMProvider {
    let providerID = "anthropic"
    let modelID: String
    let isAvailable: Bool
    let apiKey: String

    init(config: AnthropicConfig) throws {
        self.apiKey = config.apiKey
        self.modelID = config.modelID
        self.isAvailable = !apiKey.isEmpty
    }

    func generateText(prompt: String, maxTokens: Int?, temperature: Double?) async throws -> String {
        let provider = AnthropicProvider(apiKey: apiKey)
        let result = try await provider.generateText(
            model: provider.languageModel(modelID: modelID),
            prompt: prompt,
            maxTokens: maxTokens ?? 4096,
            temperature: temperature ?? 0.7
        )
        return result
    }

    func estimateCost(inputTokens: Int, outputTokens: Int) -> Decimal {
        // Claude pricing: $3.00/1M input, $15.00/1M output
        let inputCost = Decimal(inputTokens) / 1_000_000 * 3.0
        let outputCost = Decimal(outputTokens) / 1_000_000 * 15.0
        return inputCost + outputCost
    }
}

// Similar implementations for OpenAI, Google, Groq, xAI, Together

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| User → Keychain | Trusted local storage (encrypted at rest, iCloud sync) |
| App → Cloud Providers | Untrusted HTTPS (user API keys, external costs) |
| Test Call → Cloud API | Semi-trusted HTTPS (validation only, minimal cost) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-166-01 | S | API key input (user-provided) | mitigate | Validate key format before storing, test call on save per MOD-03, sanitize key from logs. |
| T-166-02 | I | Keychain storage (iCloud sync) | mitigate | Keychain encrypted at rest, iCloud sync optional (opt-out default), device-local fallback per MOD-04. |
| T-166-03 | D | Cloud API calls (user cost) | mitigate | Cost tracking per MOD-12, warn on runaway spend (> $100/message), pure BYOK (no proxy) per MOD-05. |
| T-166-04 | E | Cloud provider unavailable | mitigate | Fallback to FoundationModels per MOD-11, show banner explaining fallback, test call validates availability. |
| T-166-05 | R | API key privilege escalation | accept | API keys scoped to app usage, no sudo/root, provider-side rate limiting prevents abuse. |
| T-166-06 | T | API key telemetry (leakage) | mitigate | No API keys in logs (sanitize before logging), Keychain encrypted at rest, pure BYOK (no proxy) per MOD-05. |

## Adversarial Scenario

**Attack:** Attacker gains access to device, dumps Keychain, steals API keys, uses them for unrelated purposes. **Mitigation:** (S) Test call validates keys before storage, (I) Keychain encrypted at rest, device-local only when iCloud disabled per MOD-04, (D) Cost tracking alerts user to unusual spend, (E) Provider rate limiting limits attacker usage, (R) Keys scoped to app (not system-wide), (T) No keys in logs, sanitized output.
</threat_model>
