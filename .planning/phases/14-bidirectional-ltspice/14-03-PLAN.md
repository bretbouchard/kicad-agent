```python
def serialize_sim_command(cmd: SimulationCommand) -> str:
    """Serialize a simulation command dataclass to LTspice directive text.

    Inverse of parse_simulation_command(): produces a directive string
    that LTspice can execute and parse_simulation_command() can parse back.

    Args:
        cmd: A SimulationCommand dataclass instance.

    Returns:
        Directive text string (e.g., ".tran 0 0.001 0 1e-06").

    Raises:
        ValueError: If the command type is not recognized.
    """
    if isinstance(cmd, TranCommand):
        return f".tran {cmd.tstart} {cmd.tstop} {cmd.tstart_meas} {cmd.tstep}"
    elif isinstance(cmd, AcCommand):
        return f".ac {cmd.sweep} {cmd.npoints} {cmd.fstart} {cmd.fstop}"
    elif isinstance(cmd, DcCommand):
        return f".dc {cmd.source} {cmd.start} {cmd.stop} {cmd.step}"
    elif isinstance(cmd, NoiseCommand):
        return f".noise {cmd.output} {cmd.source} {cmd.sweep} {cmd.npoints} {cmd.fstart} {cmd.fstop}"
    elif isinstance(cmd, OpCommand):
        return ".op"
    else:
        raise ValueError(f"Unknown simulation command type: {type(cmd)}")
```

Note: Use plain float formatting (not engineering notation in output). The values stored in the dataclasses are already floats from parse_eng_value(). LTspice accepts both engineering notation ("1ms") and plain floats ("0.001"). Plain float output keeps serialize_sim_command() simple and deterministic. Round-trip validation confirms parse_simulation_command() handles plain floats correctly.

class TestSimCommandInjection:
    """Test simulation command injection into exported .asc files."""

    def test_directive_in_exported_asc(self):
        """A TranCommand injected via AscWriter appears in the exported .asc
        and is recoverable via parse_asc()."""
        # Create AscWriter with a TranCommand
        writer = AscWriter(schematic, mapper, transformer,
                          simulation_commands=[TranCommand(0, 0.001, 0, 0.000001)])
        output = writer.write(temp_path)
        result = parse_asc(output)
        assert len(result.simulation_commands) >= 1
        cmd = result.simulation_commands[0]
        assert isinstance(cmd, TranCommand)
        assert cmd.tstop == 0.001

    def test_multiple_commands_injected(self):
        """Multiple simulation commands all survive round-trip."""
        cmds = [
            TranCommand(0, 0.001, 0, 0.000001),
            AcCommand("dec", 10, 1.0, 1000000.0),
        ]
        writer = AscWriter(schematic, mapper, transformer,
                          simulation_commands=cmds)
        output = writer.write(temp_path)
        result = parse_asc(output)
        assert len(result.simulation_commands) >= 2

    def test_add_simulation_command_method(self):
        """add_simulation_command() appends commands that survive export."""
        writer = AscWriter(schematic, mapper, transformer)
        writer.add_simulation_command(OpCommand())
        output = writer.write(temp_path)
        result = parse_asc(output)
        assert any(isinstance(c, OpCommand) for c in result.simulation_commands)

class TestFullRoundTrip:
