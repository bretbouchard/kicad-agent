       ```python
       """Eagle XML format type definitions.

       Eagle schematics use XML with .sch extension. The format is
       documented by Autodesk and uses elements for libraries, parts,
       instances, nets, and wires.
       """

       from __future__ import annotations

       from dataclasses import dataclass, field
       from typing import Any, Optional


       @dataclass(frozen=True)
       class EaglePin:
           """A pin in an Eagle symbol definition."""
           name: str
           x: float = 0.0
           y: float = 0.0
           direction: str = ""     # "in", "out", "io", "pwr", "nc", "pas", "oc", "oe"
           length: str = "middle"
           visible: str = "both"


       @dataclass(frozen=True)
       class EagleSymbol:
           """A symbol definition from an Eagle library."""
           name: str
           pins: list[EaglePin] = field(default_factory=list)


       @dataclass(frozen=True)
       class EagleGate:
           """A gate in an Eagle deviceset."""
           name: str               # "A", "B", "G$1"
           symbol_name: str        # Reference to EagleSymbol
           x: float = 0.0
           y: float = 0.0


       @dataclass(frozen=True)
       class EaglePackagePad:
           """A pad in an Eagle package definition."""
           name: str               # "1", "2", "EP"
           x: float = 0.0
           y: float = 0.0
           drill: float = 0.8


       @dataclass(frozen=True)
       class EagleConnect:
           """A gate-pin to pad mapping in an Eagle device."""
           gate: str               # "A"
           pin: str                # "OUT_A"
           pad: str                # "1"


       @dataclass(frozen=True)
       class EagleDevice:
           """A device variant in an Eagle deviceset."""
           name: str               # "" (default) or specific variant
           package_name: str = ""  # Package reference
           connects: list[EagleConnect] = field(default_factory=list)


       @dataclass(frozen=True)
       class EagleDeviceset:
           """A deviceset in an Eagle library."""
           name: str               # "NE5532"
           prefix: str = ""        # "U", "R", "C"
           gates: list[EagleGate] = field(default_factory=list)
           devices: list[EagleDevice] = field(default_factory=list)


       @dataclass(frozen=True)
       class EagleLibrary:
           """An Eagle library."""
           name: str
           symbols: list[EagleSymbol] = field(default_factory=list)
           devicesets: list[EagleDeviceset] = field(default_factory=list)


       @dataclass(frozen=True)
       class EaglePart:
           """A part declaration in an Eagle schematic."""
           name: str               # "U1", "R1"
           library: str            # Library name
           deviceset: str          # "NE5532"
           device: str             # "" or specific device variant
           value: str = ""


       @dataclass(frozen=True)
       class EagleInstance:
           """A placed instance of a part on a sheet."""
           part: str               # "U1"
           gate: str               # "A", "B", "G$1"
           x: float = 0.0
           y: float = 0.0
           rotation: float = 0.0
           smashed: bool = False


       @dataclass(frozen=True)
       class EagleWire:
           """A wire segment in an Eagle net segment."""
           x1: float
           y1: float
           x2: float
           y2: float
           width: float = 0.0
           layer: str = "91"


       @dataclass(frozen=True)
       class EaglePinRef:
           """A pin reference in an Eagle net segment."""
           part: str               # "U1"
           gate: str               # "A"
           pin: str                # "OUT_A"


       @dataclass(frozen=True)
       class EagleSegment:
           """A net segment containing wires and pin references."""
           wires: list[EagleWire] = field(default_factory=list)
           pinrefs: list[EaglePinRef] = field(default_factory=list)
           labels: list[str] = field(default_factory=list)


       @dataclass(frozen=True)
       class EagleNet:
           """A net in an Eagle schematic sheet."""
           name: str
           net_class: str = "0"
           segments: list[EagleSegment] = field(default_factory=list)


       @dataclass
       class EagleSchematic:
           """Complete parsed Eagle schematic."""
           version: str = ""
           libraries: list[EagleLibrary] = field(default_factory=list)
           parts: list[EaglePart] = field(default_factory=list)
           instances: list[EagleInstance] = field(default_factory=list)
           nets: list[EagleNet] = field(default_factory=list)
       ```

       ```python
       """Eagle XML -> AbstractCircuit parser.

       Converts Eagle XML schematics (.sch) into the format-agnostic
       AbstractCircuit representation using xml.etree.ElementTree.
       """

       from __future__ import annotations

       import logging
       import xml.etree.ElementTree as ET
       from pathlib import Path
       from typing import Any, Optional

       from kicad_agent.abstract_ast.models import (
           AbstractCircuit,
           AbstractComponent,
           AbstractNet,
           AbstractPin,
           PinType,
           Position,
           WireSegment,
       )
       from kicad_agent.formats.eagle_types import (
           EagleSchematic,
           EagleLibrary,
           EagleSymbol,
           EaglePin,
           EagleDeviceset,
           EagleGate,
           EagleDevice,
           EagleConnect,
           EaglePart,
           EagleInstance,
           EagleNet,
           EagleSegment,
           EagleWire,
           EaglePinRef,
       )

       logger = logging.getLogger(__name__)

       # Eagle pin direction -> PinType mapping
       _EAGLE_DIRECTION_MAP: dict[str, PinType] = {
           "in": PinType.INPUT,
           "out": PinType.OUTPUT,
           "io": PinType.BIDI,
           "oc": PinType.OUTPUT,     # Open collector -> OUTPUT
           "oe": PinType.OUTPUT,     # Open emitter -> OUTPUT
           "pas": PinType.PASSIVE,
           "pwr": PinType.POWER_IN,  # Could be power_in or power_out, default to IN
           "nc": PinType.NO_CONNECT,
           "sup": PinType.POWER_IN,  # Supply pin
           "hiz": PinType.UNSPECIFIED,  # High impedance
           ""  : PinType.UNSPECIFIED,
       }


       class EagleParser:
           """Parse Eagle XML schematics into AbstractCircuit.

           Usage:
               parser = EagleParser()
               circuit = parser.parse("my_schematic.sch")
           """

           def parse(self, path: str | Path) -> AbstractCircuit:
               """Parse an Eagle XML schematic into AbstractCircuit.

               Args:
                   path: Path to Eagle .sch file.

               Returns:
                   AbstractCircuit with components, nets, and metadata.

               Raises:
                   FileNotFoundError: If file does not exist.
                   ValueError: If XML is not a valid Eagle schematic.
               """
               resolved = Path(path).resolve()
               if not resolved.exists():
                   raise FileNotFoundError(f"File not found: {resolved}")

               tree = ET.parse(str(resolved))
               root = tree.getroot()

               if root.tag != "eagle":
                   raise ValueError(
                       f"Not an Eagle schematic: root element is '{root.tag}', expected 'eagle'"
                   )

               version = root.get("version", "")

               # Parse the Eagle XML hierarchy
               eagle_sch = self._parse_eagle_xml(root)

               # Convert to AbstractCircuit
               components = self._convert_components(eagle_sch)
               nets = self._convert_nets(eagle_sch)

               return AbstractCircuit(
                   name=resolved.stem,
                   components=components,
                   nets=nets,
                   metadata={
                       "source_format": "eagle_xml",
                       "source_path": str(resolved),
                       "eagle_version": version,
                   },
               )

           def _parse_eagle_xml(self, root: ET.Element) -> EagleSchematic:
               """Parse the full Eagle XML hierarchy into typed structures."""
               sch = EagleSchematic(version=root.get("version", ""))

               drawing = root.find("drawing")
               if drawing is None:
                   return sch

               schematic = drawing.find("schematic")
               if schematic is None:
                   return sch

               # Parse libraries
               for lib_elem in schematic.findall(".//libraries/library"):
                   lib = self._parse_library(lib_elem)
                   sch.libraries.append(lib)

               # Parse parts
               for part_elem in schematic.findall(".//parts/part"):
                   sch.parts.append(EaglePart(
                       name=part_elem.get("name", ""),
                       library=part_elem.get("library", ""),
                       deviceset=part_elem.get("deviceset", ""),
                       device=part_elem.get("device", ""),
                       value=part_elem.get("value", ""),
                   ))

               # Parse instances and nets from sheets
               for sheet in schematic.findall(".//sheets/sheet"):
                   for inst in sheet.findall(".//instances/instance"):
                       sch.instances.append(EagleInstance(
                           part=inst.get("part", ""),
                           gate=inst.get("gate", ""),
                           x=float(inst.get("x", "0")),
                           y=float(inst.get("y", "0")),
                           rotation=self._parse_rotation(inst.get("rot", "")),
                       ))

                   for net_elem in sheet.findall(".//nets/net"):
                       sch.nets.append(self._parse_net(net_elem))

               return sch

           def _parse_library(self, lib_elem: ET.Element) -> EagleLibrary:
               """Parse a library element."""
               lib = EagleLibrary(name=lib_elem.get("name", ""))

               # Parse symbols
               for sym_elem in lib_elem.findall(".//symbols/symbol"):
                   pins = []
                   for pin_elem in sym_elem.findall("pin"):
                       pins.append(EaglePin(
                           name=pin_elem.get("name", ""),
                           x=float(pin_elem.get("x", "0")),
                           y=float(pin_elem.get("y", "0")),
                           direction=pin_elem.get("direction", ""),
                       ))
                   lib.symbols.append(EagleSymbol(
                       name=sym_elem.get("name", ""),
                       pins=pins,
                   ))

               # Parse devicesets
               for ds_elem in lib_elem.findall(".//devicesets/deviceset"):
                   gates = []
                   for gate_elem in ds_elem.findall(".//gates/gate"):
                       gates.append(EagleGate(
                           name=gate_elem.get("name", ""),
                           symbol_name=gate_elem.get("symbol", ""),
                           x=float(gate_elem.get("x", "0")),
                           y=float(gate_elem.get("y", "0")),
                       ))

                   devices = []
                   for dev_elem in ds_elem.findall(".//devices/device"):
                       connects = []
                       for conn in dev_elem.findall(".//connects/connect"):
                           connects.append(EagleConnect(
                               gate=conn.get("gate", ""),
                               pin=conn.get("pin", ""),
                               pad=conn.get("pad", ""),
                           ))
                       devices.append(EagleDevice(
                           name=dev_elem.get("name", ""),
                           package_name=dev_elem.get("package", ""),
                           connects=connects,
                       ))

                   lib.devicesets.append(EagleDeviceset(
                       name=ds_elem.get("name", ""),
                       prefix=ds_elem.get("prefix", ""),
                       gates=gates,
                       devices=devices,
                   ))

               return lib

           def _parse_net(self, net_elem: ET.Element) -> EagleNet:
               """Parse a net element."""
               segments = []
               for seg_elem in net_elem.findall("segment"):
                   wires = []
                   pinrefs = []
                   labels = []

                   for wire_elem in seg_elem.findall("wire"):
                       wires.append(EagleWire(
                           x1=float(wire_elem.get("x1", "0")),
                           y1=float(wire_elem.get("y1", "0")),
                           x2=float(wire_elem.get("x2", "0")),
                           y2=float(wire_elem.get("y2", "0")),
                       ))

                   for pinref_elem in seg_elem.findall("pinref"):
                       pinrefs.append(EaglePinRef(
                           part=pinref_elem.get("part", ""),
                           gate=pinref_elem.get("gate", ""),
                           pin=pinref_elem.get("pin", ""),
                       ))

                   for label_elem in seg_elem.findall("label"):
                       labels.append(label_elem.get("x", "") + "," + label_elem.get("y", ""))

                   segments.append(EagleSegment(wires=wires, pinrefs=pinrefs, labels=labels))

               return EagleNet(
                   name=net_elem.get("name", ""),
                   net_class=net_elem.get("class", "0"),
                   segments=segments,
               )

           def _convert_components(self, eagle_sch: EagleSchematic) -> list[AbstractComponent]:
               """Convert Eagle parts + instances to AbstractComponent.

               Multi-gate parts: All instances with the same part name
               are merged into one AbstractComponent with all pins from
               all gates.
               """
               # Build library index for pin extraction
               lib_index = self._build_library_index(eagle_sch)

               # Group instances by part name
               part_instances: dict[str, list[EagleInstance]] = {}
               for inst in eagle_sch.instances:
                   part_instances.setdefault(inst.part, []).append(inst)

               # Build part index
               part_index = {p.name: p for p in eagle_sch.parts}

               components = []
               for part_name, instances in part_instances.items():
                   part = part_index.get(part_name)
                   if not part:
                       continue

                   # Find library and deviceset for pin extraction
                   lib = next((l for l in eagle_sch.libraries if l.name == part.library), None)
                   deviceset = None
                   if lib:
                       deviceset = next(
                           (ds for ds in lib.devicesets if ds.name == part.deviceset),
                           None,
                       )

                   # Extract pins from all gate instances
                   pins = self._extract_eagle_pins(lib, deviceset, instances)

                   # Use first instance position
                   first_inst = instances[0]

                   components.append(AbstractComponent(
                       ref=part_name,
                       lib_id=f"{part.library}:{part.deviceset}" if part.library else part.deviceset,
                       value=part.value,
                       footprint=self._resolve_footprint(lib, deviceset, part.device),
                       position=Position(x=first_inst.x, y=first_inst.y),
                       rotation=first_inst.rotation,
                       pins=pins,
                       properties={
                           "eagle_library": part.library,
                           "eagle_deviceset": part.deviceset,
                           "eagle_device": part.device,
                       },
                   ))

               return components

           def _extract_eagle_pins(
               self,
               lib: Optional[EagleLibrary],
               deviceset: Optional[EagleDeviceset],
               instances: list[EagleInstance],
           ) -> list[AbstractPin]:
               """Extract pins from all gate instances of a part."""
               pins = []
               if not lib or not deviceset:
                   return pins

               # Build symbol index
               symbol_index = {s.name: s for s in lib.symbols}

               # Build connect index (gate+pin -> pad mapping)
               connect_index = {}
               for dev in deviceset.devices:
                   for conn in dev.connects:
                       connect_index[(conn.gate, conn.pin)] = conn.pad

               for inst in instances:
                   gate = next(
                       (g for g in deviceset.gates if g.name == inst.gate),
                       None,
                   )
                   if not gate:
                       continue

                   symbol = symbol_index.get(gate.symbol_name)
                   if not symbol:
                       continue

                   for sym_pin in symbol.pins:
                       # Map pin to pad number via connect index
                       pad = connect_index.get((inst.gate, sym_pin.name), sym_pin.name)
                       pin_type = _EAGLE_DIRECTION_MAP.get(sym_pin.direction, PinType.UNSPECIFIED)
                       pins.append(AbstractPin(
                           number=pad,
                           name=sym_pin.name,
                           pin_type=pin_type,
                       ))

               return pins

           def _convert_nets(self, eagle_sch: EagleSchematic) -> list[AbstractNet]:
               """Convert Eagle nets to AbstractNet."""
               nets = []
               for eagle_net in eagle_sch.nets:
                   pin_refs = []
                   wire_segments = []

                   for segment in eagle_net.segments:
                       for pinref in segment.pinrefs:
                           pin_refs.append((pinref.part, pinref.gate))
                       for wire in segment.wires:
                           wire_segments.append(WireSegment(
                               start=Position(x=wire.x1, y=wire.y1),
                               end=Position(x=wire.x2, y=wire.y2),
                           ))

                   if pin_refs or wire_segments:
                       nets.append(AbstractNet(
                           name=eagle_net.name,
                           pin_refs=pin_refs,
                           wire_segments=wire_segments,
                           labels=[eagle_net.name],
                       ))

               return nets

           def _build_library_index(self, eagle_sch: EagleSchematic) -> dict:
               """Build library lookup index."""
               return {lib.name: lib for lib in eagle_sch.libraries}

           def _resolve_footprint(
               self, lib: Optional[EagleLibrary],
               deviceset: Optional[EagleDeviceset],
               device_name: str,
           ) -> Optional[str]:
               """Resolve Eagle package to footprint string."""
               if not deviceset:
                   return None
               for dev in deviceset.devices:
                   if dev.name == device_name and dev.package_name:
                       return dev.package_name
               return None

           @staticmethod
           def _parse_rotation(rot_str: str) -> float:
               """Parse Eagle rotation string (e.g., 'R90' -> 90.0)."""
               if not rot_str:
                   return 0.0
               if rot_str.startswith("R"):
                   try:
                       return float(rot_str[1:])
                   except ValueError:
                       return 0.0
               if rot_str.startswith("M"):  # Mirror
                   return 0.0
               return 0.0
       ```

       ```python
       """AbstractCircuit -> Eagle XML writer.

       Converts the format-agnostic AbstractCircuit into Eagle XML
       schematic format (.sch).
       """

       from __future__ import annotations

       import logging
       import xml.etree.ElementTree as ET
       from pathlib import Path
       from typing import Optional

       from kicad_agent.abstract_ast.models import (
           AbstractCircuit,
           AbstractComponent,
           AbstractNet,
           AbstractPin,
           PinType,
       )

       logger = logging.getLogger(__name__)

       # PinType -> Eagle pin direction (reverse of parser mapping)
       _PINTYPE_TO_EAGLE: dict[PinType, str] = {
           PinType.INPUT: "in",
           PinType.OUTPUT: "out",
           PinType.BIDI: "io",
           PinType.PASSIVE: "pas",
           PinType.POWER_IN: "pwr",
           PinType.POWER_OUT: "pwr",
           PinType.UNSPECIFIED: "",
           PinType.NO_CONNECT: "nc",
       }


       class EagleWriter:
           """Write AbstractCircuit to Eagle XML format.

           Usage:
               writer = EagleWriter()
               writer.write(circuit, "output.sch")
           """

           def write(self, circuit: AbstractCircuit, output_path: str | Path) -> None:
               """Write an AbstractCircuit as Eagle XML.

               Args:
                   circuit: AbstractCircuit to export.
                   output_path: Destination file path.

               Raises:
                   ValueError: If circuit is empty.
               """
               if not circuit.components and not circuit.sheets:
                   raise ValueError("Cannot write empty circuit to Eagle")

               # Build Eagle XML tree
               eagle = ET.Element("eagle", version="9.6.2")
               drawing = ET.SubElement(eagle, "drawing")
               schematic = ET.SubElement(drawing, "schematic")

               # Build libraries section (single library for all components)
               libraries = ET.SubElement(schematic, "libraries")
               lib = ET.SubElement(libraries, "library", name="kicad_agent_export")
               symbols = ET.SubElement(lib, "symbols")
               devicesets = ET.SubElement(lib, "devicesets")

               # Build parts and instances
               parts = ET.SubElement(schematic, "parts")
               sheets = ET.SubElement(schematic, "sheets")
               sheet = ET.SubElement(sheets, "sheet")
               instances = ET.SubElement(sheet, "instances")
               nets_elem = ET.SubElement(sheet, "nets")

               # Convert components
               for comp in self._flatten_components(circuit):
                   self._write_component(
                       comp, parts, instances, symbols, devicesets
                   )

               # Convert nets
               for net in self._flatten_nets(circuit):
                   self._write_net(net, nets_elem)

               # Write XML
               tree = ET.ElementTree(eagle)
               ET.indent(tree, space="  ")
               tree.write(str(output_path), xml_declaration=True, encoding="utf-8")

           def _write_component(
               self,
               comp: AbstractComponent,
               parts: ET.Element,
               instances: ET.Element,
               symbols: ET.Element,
               devicesets: ET.Element,
           ) -> None:
               """Write a component as part + instance + symbol."""
               # Part declaration
               ET.SubElement(parts, "part", {
                   "name": comp.ref,
                   "library": "kicad_agent_export",
                   "deviceset": comp.lib_id,
                   "device": "",
                   "value": comp.value,
               })

               # Instance placement
               inst_attrs = {
                   "part": comp.ref,
                   "gate": "G$1",
               }
               if comp.position:
                   inst_attrs["x"] = str(comp.position.x)
                   inst_attrs["y"] = str(comp.position.y)
               if comp.rotation != 0.0:
                   inst_attrs["rot"] = f"R{comp.rotation}"
               ET.SubElement(instances, "instance", inst_attrs)

               # Symbol with pins (if component has pins)
               if comp.pins:
                   sym = ET.SubElement(symbols, "symbol", name=comp.lib_id)
                   for pin in comp.pins:
                       direction = _PINTYPE_TO_EAGLE.get(pin.pin_type, "")
                       ET.SubElement(sym, "pin", {
                           "name": pin.name,
                           "direction": direction,
                       })

               # Deviceset with single gate
               ds = ET.SubElement(devicesets, "deviceset", {
                   "name": comp.lib_id,
                   "prefix": self._extract_prefix(comp.ref),
               })
               gates = ET.SubElement(ds, "gates")
               ET.SubElement(gates, "gate", name="G$1", symbol=comp.lib_id)
               devices = ET.SubElement(ds, "devices")
               dev_attrs = {"name": ""}
               if comp.footprint:
                   dev_attrs["package"] = comp.footprint
               ET.SubElement(devices, "device", dev_attrs)

           def _write_net(self, net: AbstractNet, nets_elem: ET.Element) -> None:
               """Write a net with wire segments."""
               net_elem = ET.SubElement(nets_elem, "net", {
                   "name": net.name,
                   "class": "0",
               })

               if net.wire_segments or net.pin_refs:
                   seg = ET.SubElement(net_elem, "segment")

                   for wire in net.wire_segments:
                       ET.SubElement(seg, "wire", {
                           "x1": str(wire.start.x),
                           "y1": str(wire.start.y),
                           "x2": str(wire.end.x),
                           "y2": str(wire.end.y),
                           "width": "0",
                           "layer": "91",
                       })

                   for ref, pin_number in net.pin_refs:
                       ET.SubElement(seg, "pinref", {
                           "part": ref,
                           "gate": "G$1",
                           "pin": pin_number,
                       })

           def _flatten_components(self, circuit: AbstractCircuit) -> list[AbstractComponent]:
               """Get all components from circuit and sheets."""
               components = list(circuit.components)
               for sheet in circuit.sheets:
                   components.extend(sheet.components)
               return components

           def _flatten_nets(self, circuit: AbstractCircuit) -> list[AbstractNet]:
               """Get all nets from circuit and sheets."""
               nets = list(circuit.nets)
               for sheet in circuit.sheets:
                   nets.extend(sheet.nets)
               return nets

           @staticmethod
           def _extract_prefix(ref: str) -> str:
               """Extract component prefix from reference (U1 -> U, R3 -> R)."""
               prefix = ""
               for ch in ref:
                   if ch.isalpha() or ch == "#":
                       prefix += ch
                   else:
                       break
               return prefix or "X"
       ```
