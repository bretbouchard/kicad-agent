       ```python
       """Format Registry: central system for multi-format adapter management.

       Provides:
       - Format registration (parser + writer + capabilities)
       - Format lookup by name
       - Capability matrix (which formats support read, write, round-trip, etc.)
       - Unified conversion pipeline
       """

       from __future__ import annotations

       import logging
       from dataclasses import dataclass
       from enum import Enum
       from typing import Any, Callable, Optional, Protocol

       from kicad_agent.abstract_ast.models import AbstractCircuit

       logger = logging.getLogger(__name__)


       class FormatType(str, Enum):
           """Supported EDA format types."""
           KICAD = "kicad"
           EASYEDA = "easyeda"
           ALTIUM = "altium"
           EAGLE = "eagle"
           OPENWATER = "openwater"


       @dataclass(frozen=True)
       class FormatCapabilities:
           """Capabilities of a format adapter."""
           read: bool
           write: bool
           round_trip: bool
           schematic: bool
           pcb: bool
           hierarchy: bool       # Multi-sheet support
           component_libs: bool  # Library access


       class Parser(Protocol):
           """Protocol for format parsers."""
           def parse(self, path: str) -> AbstractCircuit: ...


       class Writer(Protocol):
           """Protocol for format writers."""
           def write(self, circuit: AbstractCircuit, output_path: str) -> None: ...


       @dataclass
       class FormatAdapter:
           """A registered format adapter with parser, writer, and capabilities."""
           format_type: FormatType
           parser: type
           writer: Optional[type]
           capabilities: FormatCapabilities


       class FormatRegistry:
           """Central registry for format adapters.

           Usage:
               reg = FormatRegistry()
               parser_cls = reg.get_parser("eagle")
               parser = parser_cls()
               circuit = parser.parse("input.sch")

               # Or auto-detect:
               fmt = detect_format("input.sch")
               parser_cls = reg.get_parser(fmt)
           """

           _adapters: dict[FormatType, FormatAdapter]

           def __init__(self) -> None:
               self._adapters = {}
               self._register_builtin_formats()

           def register(
               self,
               format_type: FormatType,
               parser: type,
               writer: Optional[type],
               capabilities: FormatCapabilities,
           ) -> None:
               """Register a format adapter.

               Args:
                   format_type: The format identifier.
                   parser: Parser class with parse() method.
                   writer: Writer class with write() method, or None for read-only.
                   capabilities: FormatCapabilities describing what the adapter supports.
               """
               self._adapters[format_type] = FormatAdapter(
                   format_type=format_type,
                   parser=parser,
                   writer=writer,
                   capabilities=capabilities,
               )
               logger.info(f"Registered format adapter: {format_type.value}")

           def get_parser(self, format_type: str | FormatType) -> Optional[type]:
               """Get parser class for a format."""
               fmt = FormatType(format_type) if isinstance(format_type, str) else format_type
               adapter = self._adapters.get(fmt)
               return adapter.parser if adapter else None

           def get_writer(self, format_type: str | FormatType) -> Optional[type]:
               """Get writer class for a format, or None if read-only."""
               fmt = FormatType(format_type) if isinstance(format_type, str) else format_type
               adapter = self._adapters.get(fmt)
               return adapter.writer if adapter else None

           def capability_matrix(self) -> dict[str, FormatCapabilities]:
               """Get capabilities for all registered formats."""
               return {
                   fmt.value: adapter.capabilities
                   for fmt, adapter in self._adapters.items()
               }

           def registered_formats(self) -> list[str]:
               """Get list of registered format names."""
               return sorted(fmt.value for fmt in self._adapters.keys())

           def convert(
               self,
               input_path: str,
               output_path: str,
               from_format: str | FormatType,
               to_format: str | FormatType,
           ) -> AbstractCircuit:
               """Convert a file from one format to another.

               Reads input with from_format parser -> AbstractCircuit
               Writes AbstractCircuit with to_format writer -> output file.

               Returns the intermediate AbstractCircuit.
               """
               from_fmt = FormatType(from_format) if isinstance(from_format, str) else from_format
               to_fmt = FormatType(to_format) if isinstance(to_format, str) else to_format

               # Parse
               parser_cls = self.get_parser(from_fmt)
               if not parser_cls:
                   raise ValueError(f"No parser registered for format: {from_fmt.value}")

               parser = parser_cls()
               circuit = parser.parse(input_path)

               # Write
               writer_cls = self.get_writer(to_fmt)
               if not writer_cls:
                   raise ValueError(f"No writer registered for format: {to_fmt.value}")

               writer = writer_cls()
               writer.write(circuit, output_path)

               return circuit

           def _register_builtin_formats(self) -> None:
               """Register all built-in format adapters."""
               # KiCad
               self.register(
                   FormatType.KICAD,
                   parser=self._lazy_import("kicad_agent.abstract_ast.kicad_adapter", "KiCadAdapter"),
                   writer=self._lazy_import("kicad_agent.abstract_ast.kicad_adapter", "KiCadAdapter"),
                   capabilities=FormatCapabilities(
                       read=True, write=True, round_trip=True,
                       schematic=True, pcb=True, hierarchy=True,
                       component_libs=True,
                   ),
               )

               # EasyEDA
               self.register(
                   FormatType.EASYEDA,
                   parser=self._lazy_import("kicad_agent.formats.easyeda_parser", "EasyEdaParser"),
                   writer=self._lazy_import("kicad_agent.formats.easyeda_writer", "EasyEdaWriter"),
                   capabilities=FormatCapabilities(
                       read=True, write=True, round_trip=True,
                       schematic=True, pcb=False, hierarchy=False,
                       component_libs=True,
                   ),
               )

               # Altium (read-only)
               self.register(
                   FormatType.ALTIUM,
                   parser=self._lazy_import("kicad_agent.formats.altium_parser", "AltiumParser"),
                   writer=None,
                   capabilities=FormatCapabilities(
                       read=True, write=False, round_trip=False,
                       schematic=True, pcb=False, hierarchy=True,
                       component_libs=True,
                   ),
               )

               # Eagle
               self.register(
                   FormatType.EAGLE,
                   parser=self._lazy_import("kicad_agent.formats.eagle_parser", "EagleParser"),
                   writer=self._lazy_import("kicad_agent.formats.eagle_writer", "EagleWriter"),
                   capabilities=FormatCapabilities(
                       read=True, write=True, round_trip=True,
                       schematic=True, pcb=False, hierarchy=False,
                       component_libs=True,
                   ),
               )

               # OpenWater (proof of concept)
               self.register(
                   FormatType.OPENWATER,
                   parser=self._lazy_import("kicad_agent.formats.openwater_parser", "OpenWaterParser"),
                   writer=None,
                   capabilities=FormatCapabilities(
                       read=True, write=False, round_trip=False,
                       schematic=True, pcb=False, hierarchy=False,
                       component_libs=False,
                   ),
               )

           @staticmethod
           def _lazy_import(module: str, class_name: str) -> type:
               """Create a lazy import proxy that defers actual import until first use.

               Returns a class that imports and delegates on first instantiation.
               """
               import importlib

               class _LazyProxy:
                   _real_class: type | None = None

                   @classmethod
                   def _resolve(cls) -> type:
                       if cls._real_class is None:
                           mod = importlib.import_module(module)
                           cls._real_class = getattr(mod, class_name)
                       return cls._real_class

                   def __new__(cls, *args, **kwargs):
                       real = cls._resolve()
                       return real(*args, **kwargs)

               _LazyProxy.__name__ = class_name
               _LazyProxy.__qualname__ = class_name
               return _LazyProxy
       ```

       ```python
       """Auto-detect EDA format from file extension and content.

       Detection strategy:
       1. Extension-based: .kicad_sch -> kicad, .SchDoc -> altium
       2. Content-based: JSON with head.docType -> easyeda, XML with <eagle> -> eagle
       3. Fallback: raise ValueError for unknown formats
       """

       from __future__ import annotations

       import json
       import logging
       from pathlib import Path
       from typing import Optional

       logger = logging.getLogger(__name__)


       # Extension -> format mapping (checked first)
       _EXTENSION_MAP: dict[str, str] = {
           ".kicad_sch": "kicad",
           ".kicad_pcb": "kicad",
           ".kicad_sym": "kicad",
           ".kicad_mod": "kicad",
           ".schdoc": "altium",
           ".pcbdoc": "altium",
       }


       def detect_format(path: str | Path) -> str:
           """Auto-detect EDA format from file extension and content.

           Args:
               path: Path to the EDA file.

           Returns:
               Format type string: "kicad", "easyeda", "altium", "eagle", "openwater".

           Raises:
               ValueError: If format cannot be detected.
           """
           resolved = Path(path)
           suffix = resolved.suffix.lower()

           # 1. Check definitive extensions
           if suffix in _EXTENSION_MAP:
               return _EXTENSION_MAP[suffix]

           # 2. Content-based detection for ambiguous extensions
           if suffix == ".json":
               return _detect_json_format(resolved)

           if suffix == ".sch" or suffix == ".brd":
               return _detect_sch_format(resolved)

           # 3. Try content-based detection as last resort
           try:
               return _detect_by_content(resolved)
           except Exception:
               pass

           raise ValueError(
               f"Cannot detect format for: {path}. "
               f"Supported formats: kicad, easyeda, altium, eagle"
           )


       def _detect_json_format(path: Path) -> str:
           """Detect format for .json files (EasyEDA or generic)."""
           try:
               with open(path) as f:
                   data = json.load(f)
               if isinstance(data, dict) and "head" in data:
                   return "easyeda"
           except (json.JSONDecodeError, UnicodeDecodeError):
               pass
           raise ValueError(f"Cannot detect JSON format for: {path}")


       def _detect_sch_format(path: Path) -> str:
           """Detect format for .sch files (Eagle XML or other)."""
           try:
               with open(path) as f:
                   first_line = f.readline().strip()
               if first_line.startswith("<?xml") or first_line.startswith("<!DOCTYPE"):
                   return "eagle"
           except (UnicodeDecodeError, OSError):
               pass
           raise ValueError(f"Cannot detect .sch format for: {path}")


       def _detect_by_content(path: Path) -> str:
           """Detect format by reading file content."""
           try:
               import olefile
               if olefile.isOleFile(str(path)):
                   return "altium"
           except ImportError:
               pass

           try:
               with open(path) as f:
                   content = f.read(1024)
               if "<eagle" in content:
                   return "eagle"
               if content.strip().startswith("("):
                   return "kicad"  # S-expression
           except (UnicodeDecodeError, OSError):
               pass

           raise ValueError(f"Cannot detect format by content for: {path}")
       ```

       ```python
       """Unified CLI for multi-format EDA conversion.

       Usage:
           kicad-agent convert --from eagle --to kicad input.sch output.kicad_sch
           kicad-agent convert --from easyeda --to eagle project.json output.sch
           kicad-agent formats  # List supported formats and capabilities
       """

       from __future__ import annotations

       import argparse
       import sys

       from kicad_agent.formats.registry import FormatRegistry


       def main(argv: list[str] | None = None) -> None:
           """CLI entry point."""
           parser = argparse.ArgumentParser(
               prog="kicad-agent",
               description="Multi-format EDA conversion tool",
           )
           subparsers = parser.add_subparsers(dest="command")

           # convert subcommand
           convert_parser = subparsers.add_parser("convert", help="Convert between EDA formats")
           convert_parser.add_argument("--from", dest="from_format", required=True,
                                       help="Source format (kicad, easyeda, altium, eagle)")
           convert_parser.add_argument("--to", dest="to_format", required=True,
                                       help="Target format (kicad, easyeda, eagle)")
           convert_parser.add_argument("input", help="Input file path")
           convert_parser.add_argument("output", help="Output file path")

           # formats subcommand
           formats_parser = subparsers.add_parser("formats",
                                                   help="List supported formats")

           args = parser.parse_args(argv)

           if args.command == "convert":
               _handle_convert(args)
           elif args.command == "formats":
               _handle_formats()
           else:
               parser.print_help()


       def _handle_convert(args: argparse.Namespace) -> None:
           """Handle the convert subcommand."""
           registry = FormatRegistry()

           try:
               circuit = registry.convert(
                   args.input,
                   args.output,
                   from_format=args.from_format,
                   to_format=args.to_format,
               )
               print(f"Converted {args.input} ({args.from_format}) -> {args.output} ({args.to_format})")
               print(f"  Components: {len(circuit.components)}")
               print(f"  Nets: {len(circuit.nets)}")
           except ValueError as e:
               print(f"Error: {e}", file=sys.stderr)
               sys.exit(1)


       def _handle_formats() -> None:
           """Handle the formats subcommand."""
           registry = FormatRegistry()
           matrix = registry.capability_matrix()

           print("Supported formats:")
           print(f"{'Format':<12} {'Read':>5} {'Write':>5} {'R/T':>5} {'Sch':>5} {'PCB':>5} {'Hier':>5}")
           print("-" * 50)
           for name, caps in sorted(matrix.items()):
               print(
                   f"{name:<12} "
                   f"{'Y' if caps.read else 'N':>5} "
                   f"{'Y' if caps.write else 'N':>5} "
                   f"{'Y' if caps.round_trip else 'N':>5} "
                   f"{'Y' if caps.schematic else 'N':>5} "
                   f"{'Y' if caps.pcb else 'N':>5} "
                   f"{'Y' if caps.hierarchy else 'N':>5}"
               )


       if __name__ == "__main__":
           main()
       ```
