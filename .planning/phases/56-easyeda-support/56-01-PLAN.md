       """EasyEDA JSON format type definitions.

       EasyEDA schematics are JSON with a shape array containing tilde-delimited
       entries. Each entry type is identified by its prefix (LIB, W, L, J, etc.)

       Schema reference: https://docs.easyeda.com/en/DocumentFormat/
       """

       from __future__ import annotations

       from dataclasses import dataclass, field
       from typing import Any, Optional


       @dataclass(frozen=True)
       class EasyEdaHead:
           """EasyEDA document header."""
           doc_type: str           # "1" for schematic
           editor_version: str

           @classmethod
           def from_dict(cls, d: dict) -> "EasyEdaHead":
               return cls(
                   doc_type=d.get("docType", ""),
                   editor_version=d.get("editorVersion", ""),
               )


       @dataclass(frozen=True)
       class EasyEdaComponent:
           """A component (LIB~ entry) parsed from EasyEDA shape array."""
           x: float
           y: float
           package: str            # EasyEDA package name
           lcsc_part: str          # LCSC part number (e.g., "C12345")
           component_name: str     # Component name (e.g., "NE5532")
           ref: str                # Reference designator (e.g., "U1")
           value: str              # Component value
           rotation: float = 0.0
           raw_fields: tuple[str, ...] = ()  # All tilde-split fields


       @dataclass(frozen=True)
       class EasyEdaWire:
           """A wire (W~ entry) parsed from EasyEDA shape array."""
           points: list[tuple[float, float]]  # List of (x, y) waypoints
           raw_fields: tuple[str, ...] = ()


       @dataclass(frozen=True)
       class EasyEdaLabel:
           """A net label (L~ entry) parsed from EasyEDA shape array."""
           x: float
           y: float
           text: str               # Net name
           raw_fields: tuple[str, ...] = ()


       @dataclass(frozen=True)
       class EasyEdaJunction:
           """A wire junction (J~ entry) parsed from EasyEDA shape array."""
           x: float
           y: float
           raw_fields: tuple[str, ...] = ()


       @dataclass
       class EasyEdaSchematic:
           """Complete parsed EasyEDA schematic."""
           head: EasyEdaHead
           components: list[EasyEdaComponent] = field(default_factory=list)
           wires: list[EasyEdaWire] = field(default_factory=list)
           labels: list[EasyEdaLabel] = field(default_factory=list)
           junctions: list[EasyEdaJunction] = field(default_factory=list)

           @classmethod
           def from_json(cls, data: dict[str, Any]) -> "EasyEdaSchematic":
               """Parse raw JSON dict into typed EasyEdaSchematic."""
               head = EasyEdaHead.from_dict(data.get("head", {}))
               shapes = data.get("shape", [])

               components = []
               wires = []
               labels = []
               junctions = []

               for shape in shapes:
                   if not isinstance(shape, str):
                       continue
                   parts = shape.split("~")
                   prefix = parts[0] if parts else ""

                   if prefix == "LIB":
                       components.append(_parse_lib_entry(parts))
                   elif prefix == "W":
                       wires.append(_parse_wire_entry(parts))
                   elif prefix == "L":
                       labels.append(_parse_label_entry(parts))
                   elif prefix == "J":
                       junctions.append(_parse_junction_entry(parts))

               return cls(
                   head=head,
                   components=components,
                   wires=wires,
                   labels=labels,
                   junctions=junctions,
               )
       ```

       """EasyEDA JSON -> AbstractCircuit parser.

       Converts EasyEDA's shape-based JSON format into the format-agnostic
       AbstractCircuit representation.
       """

       from __future__ import annotations

       import json
       import logging
       from pathlib import Path

       from kicad_agent.abstract_ast.models import (
           AbstractCircuit,
           AbstractComponent,
           AbstractNet,
           AbstractPin,
           PinType,
           Position,
           WireSegment,
       )
       from kicad_agent.formats.easyeda_types import EasyEdaSchematic

       logger = logging.getLogger(__name__)


       class EasyEdaParser:
           """Parse EasyEDA JSON schematics into AbstractCircuit.

           Usage:
               parser = EasyEdaParser()
               circuit = parser.parse("my_schematic.json")
           """

           def parse(self, path: str | Path) -> AbstractCircuit:
               """Parse an EasyEDA JSON file into AbstractCircuit.

               Args:
                   path: Path to EasyEDA JSON file.

               Returns:
                   AbstractCircuit with components, nets, and metadata.

               Raises:
                   FileNotFoundError: If file does not exist.
                   ValueError: If JSON is not a valid EasyEDA schematic.
               """
               resolved = Path(path).resolve()
               if not resolved.exists():
                   raise FileNotFoundError(f"File not found: {resolved}")

               with open(resolved) as f:
                   data = json.load(f)

               if "head" not in data:
                   raise ValueError("Not a valid EasyEDA file: missing 'head' field")

               easyeda = EasyEdaSchematic.from_json(data)

               components = self._convert_components(easyeda)
               nets = self._derive_nets(easyeda)

               return AbstractCircuit(
                   name=resolved.stem,
                   components=components,
                   nets=nets,
                   metadata={
                       "source_format": "easyeda_json",
                       "source_path": str(resolved),
                       "easyeda_version": easyeda.head.editor_version,
                   },
               )

           def _convert_components(
               self, easyeda: EasyEdaSchematic
           ) -> list[AbstractComponent]:
               """Convert EasyEdaComponent list to AbstractComponent list."""
               components = []
               for comp in easyeda.components:

                       ref=comp.ref,

       """AbstractCircuit -> EasyEDA JSON writer.

       Converts the format-agnostic AbstractCircuit back into EasyEDA's
       shape-based JSON format.
       """

       from __future__ import annotations

       import json
       import logging
       from pathlib import Path

       from kicad_agent.abstract_ast.models import (
           AbstractCircuit,
           AbstractComponent,
           AbstractNet,
       )

       logger = logging.getLogger(__name__)


       class EasyEdaWriter:
           """Write AbstractCircuit to EasyEDA JSON format.

           Usage:
               writer = EasyEdaWriter()
               writer.write(circuit, "output.json")
           """

           def write(self, circuit: AbstractCircuit, output_path: str | Path) -> None:
               """Write an AbstractCircuit as EasyEDA JSON.

               Args:
                   circuit: AbstractCircuit to export.
                   output_path: Destination file path.

               Raises:
                   ValueError: If circuit is empty.
               """
               if not circuit.components and not circuit.sheets:
                   raise ValueError("Cannot write empty circuit to EasyEDA")

               shapes: list[str] = []

               # Convert components to LIB~ entries
               for comp in self._flatten_components(circuit):
                   shapes.append(self._component_to_lib(comp))

               # Convert nets to W~ and L~ entries
               for net in self._flatten_nets(circuit):
                   for segment in net.wire_segments:
                       shapes.append(self._wire_segment_to_w(segment))
                   for label in net.labels:
                       shapes.append(self._label_to_l(label, net))

               document = {
                   "head": {
                       "docType": "1",
                       "editorVersion": "kicad-agent-export",
                   },
                   "canvas": "CA~1000~1000~10~10~1~0~0",
                   "shape": shapes,
               }

               Path(output_path).write_text(json.dumps(document, indent=2))

           def _component_to_lib(self, comp: AbstractComponent) -> str:
               """Convert AbstractComponent to LIB~ shape entry."""
               x = comp.position.x if comp.position else 0.0
               y = comp.position.y if comp.position else 0.0
               lcsc = comp.properties.get("lcsc_part", "")
               name = comp.properties.get("component_name", comp.value)
               fields = [
                   "LIB",
                   str(int(x)),
                   str(int(y)),
                   comp.lib_id,      # package
                   lcsc,              # LCSC part number
                   "",                # supplier
                   name,              # component name
                   "1",               # count
                   comp.ref,          # reference
                   str(int(comp.rotation)),
                   "0",               # flip
                   "0",               # flag
               ]
               return "~".join(fields)

           def _wire_segment_to_w(self, segment) -> str:
               """Convert WireSegment to W~ shape entry."""
               return f"W~{int(segment.start.x)}~{int(segment.start.y)}~{int(segment.end.x)}~{int(segment.end.y)}~0~"

           def _label_to_l(self, label_text: str, net: AbstractNet) -> str:
               """Convert net label to L~ shape entry."""
               # Place label at first wire segment start or (0,0)
               if net.wire_segments:
                   pos = net.wire_segments[0].start
               else:
                   pos = None
               x = int(pos.x) if pos else 0
               y = int(pos.y) if pos else 0
               return f"L~{x}~{y}~{label_text}~0~"

           def _flatten_components(self, circuit: AbstractCircuit) -> list[AbstractComponent]:
               """Get all components from circuit and sheets."""
               components = list(circuit.components)
               for sheet in circuit.sheets:
                   components.extend(sheet.components)
               return components

           def _flatten_nets(self, circuit: AbstractCircuit) -> list[AbstractNet]:
               """Get all nets from circuit and sheets."""
               nets = list(circuit.nets)
               for sheet in circuit.sheets:
                   nets.extend(sheet.nets)
               return nets
       ```
