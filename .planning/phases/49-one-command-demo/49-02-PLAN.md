       """Auto-generated documentation for demo templates."""

       from __future__ import annotations

       from kicad_agent.demo.templates import BUILTIN_TEMPLATES, list_templates


       def generate_template_docs() -> str:
           """Generate Markdown documentation for all built-in templates.

           Returns:
               Markdown string documenting each template with metadata.
           """
           lines = [
               "# Demo Templates",
               "",
               "Built-in circuit templates for `kicad-agent demo`.",
               "",
               "## Usage",
               "",
               "```bash",
               "kicad-agent demo --template <name>",
               "kicad-agent demo --template random",
               "kicad-agent demo --list",
               "```",
               "",
           ]

           tier_order = {"basic": 0, "intermediate": 1, "advanced": 2}
           tier_labels = {"basic": "Basic", "intermediate": "Intermediate", "advanced": "Advanced"}
           tier_descriptions = {
               "basic": "Simple circuits with 1-5 components. Good for testing and learning.",
               "intermediate": "Moderate circuits with 6-12 components. Op-amp filters and amplifiers.",
               "advanced": "Complex circuits with 13+ components. Multi-IC designs.",
           }

           grouped: dict[str, list] = {}
           for t in BUILTIN_TEMPLATES.values():
               grouped.setdefault(t.difficulty, []).append(t)

           for tier in ("basic", "intermediate", "advanced"):
               templates = grouped.get(tier, [])
               if not templates:
                   continue
               lines.append(f"## {tier_labels[tier]}")
               lines.append("")
               lines.append(tier_descriptions[tier])
               lines.append("")
               for t in sorted(templates, key=lambda x: x.name):
                   lines.append(f"### `{t.name}`")
                   lines.append("")
                   lines.append(f"{t.description}.")
                   lines.append("")
                   lines.append(f"- **Components:** {t.expected_component_count}")
                   lines.append(f"- **Nets:** {t.expected_net_count}")
                   lines.append(f"- **Usage:** `kicad-agent demo --template {t.name}`")
                   lines.append("")

           return "\n".join(lines)
       ```
