       class RegressionReport(BaseModel):
           current: BenchmarkResult
           baseline: BenchmarkResult
           delta: dict[str, float]            # category -> accuracy delta (current - baseline)
           overall_delta: float
           is_regression: bool
           regression_categories: list[str]   # Categories that dropped > threshold

       class RegressionDetector:
           def __init__(self, results_dir: str = "benchmarks/results", threshold: float = 0.02):
               self.results_dir = Path(results_dir)
               self.threshold = threshold

           def compare(self, current: BenchmarkResult, baseline: BenchmarkResult) -> RegressionReport:
               """Compare current result against baseline."""
               delta = {}
               regression_categories = []
               all_categories = set(list(baseline.category_accuracy.keys()) + list(current.category_accuracy.keys()))
               for cat in all_categories:
                   base_acc = baseline.category_accuracy.get(cat, 0.0)
                   curr_acc = current.category_accuracy.get(cat, 0.0)
                   diff = curr_acc - base_acc
                   delta[cat] = round(diff, 4)
                   if diff < -self.threshold:
                       regression_categories.append(cat)
               return RegressionReport(
                   current=current, baseline=baseline, delta=delta,
                   overall_delta=round(current.accuracy - baseline.accuracy, 4),
                   is_regression=len(regression_categories) > 0,
                   regression_categories=regression_categories,
               )

           def store_result(self, result: BenchmarkResult) -> Path:
               """Store result with timestamp filename."""
               self.results_dir.mkdir(parents=True, exist_ok=True)
               ts = result.evaluated_at.replace(":", "-").replace(".", "-")
               path = self.results_dir / f"{result.model_name}_{ts}.json"
               path.write_text(result.model_dump_json(indent=2))
               return path

           def load_history(self) -> list[BenchmarkResult]:
               """Load all stored results sorted by date."""
               results = []
               for path in sorted(self.results_dir.glob("*.json")):
                   results.append(BenchmarkResult.model_validate_json(path.read_text()))
               return results

           def load_baseline(self) -> BenchmarkResult:
               """Load baseline result."""
               return BenchmarkResult.model_validate_json(
                   (self.results_dir / "baseline.json").read_text()
               )

           def set_baseline(self, result: BenchmarkResult) -> None:
               """Update baseline with given result."""
               self.results_dir.mkdir(parents=True, exist_ok=True)
               (self.results_dir / "baseline.json").write_text(result.model_dump_json(indent=2))
       ```

       on:
         pull_request:

           runs-on: ubuntu-latest
           steps:
             - uses: actions/checkout@v4
             - uses: actions/setup-python@v5
               with:
                 python-version: '3.11'
             - run: pip install -e ".[dev]"
             - name: Run benchmark
               run: |
                 python -m kicad_agent.benchmarks \
                   --dataset benchmarks/pcb-mmlu-v1.json \
                   --model heuristic \
                   --output /tmp/current-results.json
             - name: Check regression
               run: |
                 python -c "
                 from kicad_agent.benchmarks.regression import RegressionDetector
                 from kicad_agent.benchmarks.runner import BenchmarkResult
                 import json
                 detector = RegressionDetector()
                 baseline = detector.load_baseline()
                 current = BenchmarkResult.model_validate(json.load(open('/tmp/current-results.json')))
                 report = detector.compare(current, baseline)
                 if report.is_regression:
                     print(f'REGRESSION DETECTED in: {report.regression_categories}')
                     print(f'Delta: {report.delta}')
                     exit(1)
                 print(f'OK: {current.accuracy:.1%} accuracy')
                 for cat, acc in current.category_accuracy.items():
                     delta = report.delta.get(cat, 0)
                     print(f'  {cat}: {acc:.1%} ({delta:+.1%})')
                 "
       ```
