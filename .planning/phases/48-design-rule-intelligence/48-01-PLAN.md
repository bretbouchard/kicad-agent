       """Design rule schemas and ABC for pluggable rule engine.

       DOMAIN-04: Domain-specific DRC beyond KiCad ERC/DRC.

       Defines the contract for design rules:
       - DesignRule ABC: name, category, severity, check() method
       - DesignRuleViolation: structured violation output
       - DesignRuleReport: aggregated report with summary stats
       - RuleSeverity: severity enum
       - RuleCategory: category enum

       Security:
         T-48-01: Violations list capped at 500 per report (DoS prevention).
         T-48-02: suggestion text max 1000 chars.
       """
       from __future__ import annotations

       from abc import ABC, abstractmethod
       from enum import Enum
       from typing import Any

       from pydantic import BaseModel, Field, field_validator


       class RuleSeverity(str, Enum):
           """Design rule violation severity."""
           INFO = "INFO"
           SUGGESTION = "SUGGESTION"
           WARNING = "WARNING"
           CRITICAL = "CRITICAL"


       class RuleCategory(str, Enum):
           """Design rule category for grouping."""
           BYPASS_CAPS = "bypass_caps"
           FEEDBACK = "feedback"
           IMPEDANCE = "impedance"
           THERMAL = "thermal"
           GROUND = "ground"
           POWER = "power"
           SIGNAL = "signal"
           LAYOUT = "layout"


       class DesignRuleViolation(BaseModel):
           """A single design rule violation.

           Attributes:
               rule_id: Rule identifier (e.g. "BYPASS_CAP_01").
               description: What was found and why it matters.
               severity: Violation severity level.
               location: Where in the design (component ref, net name, or position).
               suggestion: Concrete fix recommendation.
               affected_components: Component refs involved.
               details: Additional context (optional).
           """
           rule_id: str = Field(min_length=1, max_length=64, pattern=r"^[A-Z][A-Z0-9_]*\d{2}$")
           description: str = Field(min_length=1, max_length=2000)
           severity: RuleSeverity
           location: str = Field(min_length=1, max_length=512)
           suggestion: str = Field(default="", max_length=1000)
           affected_components: tuple[str, ...] = Field(default_factory=tuple)
           details: dict[str, Any] = Field(default_factory=dict)

           @field_validator("rule_id")
           @classmethod
           def _rule_id_format(cls, v: str) -> str:
               if not v.strip():
                   raise ValueError("rule_id must not be empty")
               return v


       class DesignRuleReport(BaseModel):
           """Aggregated report from design rule checks.

           Attributes:
               violations: All violations found, sorted by severity.
               schematic_path: Path to the checked schematic.
               rules_run: Number of rules that were executed.
               rules_passed: Number of rules with no violations.
               rules_failed: Number of rules with violations.
               summary: Severity counts.
               elapsed_ms: Total execution time.
           """
           violations: tuple[DesignRuleViolation, ...] = Field(
               default_factory=tuple, max_length=500,
           )
           schematic_path: str = Field(default="")
           rules_run: int = Field(default=0, ge=0)
           rules_passed: int = Field(default=0, ge=0)
           rules_failed: int = Field(default=0, ge=0)
           summary: dict[str, int] = Field(default_factory=dict)
           elapsed_ms: float = Field(default=0.0, ge=0.0)

           def model_post_init(self, __context) -> None:
               counts = {s.value: 0 for s in RuleSeverity}
               for v in self.violations:
                   counts[v.severity.value] += 1
               self.summary = counts

           """Abstract base class for design rules.

           Subclass this to create custom rules. Register with
           DesignRuleEngine to run them against circuit topologies.

           Attributes:
               name: Rule identifier (e.g. "BYPASS_CAP_01").
               category: Rule category for grouping.
               default_severity: Default severity for violations.
               description: What this rule checks.
           """

           name: str
           category: RuleCategory
           default_severity: RuleSeverity
           description: str = ""

           def check(
               self,
               topology: Any,  # CircuitTopology from Phase 46
               config: dict[str, Any] | None = None,
           ) -> list[DesignRuleViolation]:
               """Check the topology for this rule's violations.

               Args:
                   topology: CircuitTopology to check.
                   config: Optional configuration overrides (thresholds, etc.).

               Returns:
                   List of violations found. Empty list if no violations.
               """

       ```python
       """Design rule engine -- orchestrates rule execution and report generation.

       DOMAIN-04: Runs pluggable design rules against circuit topologies.

       Usage:
           from kicad_agent.analysis.design_rule_engine import DesignRuleEngine
           from kicad_agent.analysis.builtin_rules import get_builtin_rules

           engine = DesignRuleEngine(rules=get_builtin_rules())
           report = engine.run(topology)
           for v in report.violations:
               print(f"[{v.severity.value}] {v.rule_id}: {v.description}")
       """
       from __future__ import annotations

       import logging
       import time
       from typing import Any

       from kicad_agent.analysis.design_rules import (
           DesignRule,
           DesignRuleReport,
           DesignRuleViolation,
           RuleSeverity,
       )

       logger = logging.getLogger(__name__)


       class DesignRuleEngine:
           """Orchestrates design rule execution.

           Loads rules, runs them against a topology, and produces
           a structured report with all violations.

           Args:
               rules: List of DesignRule instances to run.
               disabled_rules: Set of rule names to skip.
               config: Per-rule configuration overrides.

           Usage:
               engine = DesignRuleEngine(rules=get_builtin_rules())
               report = engine.run(topology)
           """

           def __init__(
               self,
               rules: list[DesignRule] | None = None,
               disabled_rules: set[str] | None = None,
               config: dict[str, dict[str, Any]] | None = None,
           ):
               self._rules = rules or []
               self._disabled = disabled_rules or set()
               self._config = config or {}

           def run(self, topology: Any) -> DesignRuleReport:
               """Run all enabled rules against the topology.

               Algorithm:
               1. Filter out disabled rules
               2. For each enabled rule, call check() with config
               3. Collect violations, handle errors gracefully
               4. Sort violations by severity (CRITICAL first)
               5. Build and return DesignRuleReport

               Error handling: if a rule raises an exception, log it
               and continue with remaining rules. Never let one broken
               rule kill the entire check.

               Args:
                   topology: CircuitTopology from Phase 46.

               Returns:
                   DesignRuleReport with all violations and summary.
               """
               start = time.monotonic()
               all_violations: list[DesignRuleViolation] = []
               rules_run = 0
               rules_passed = 0
               rules_failed = 0

               for rule in self._rules:
                   if rule.name in self._disabled:
                       logger.debug("Skipping disabled rule: %s", rule.name)
                       continue

                   rules_run += 1
                   rule_config = self._config.get(rule.name, {})

                   try:
                       violations = rule.check(topology, config=rule_config)
                   except Exception as e:
                       logger.error(
                           "Rule %s raised exception: %s", rule.name, e,
                           exc_info=True,
                       )
                       # Create a meta-violation for the broken rule
                       violations = [DesignRuleViolation(
                           rule_id=rule.name,
                           description=f"Rule execution failed: {e}",
                           severity=RuleSeverity.WARNING,
                           location="(rule engine)",
                           suggestion=f"Report this as a bug: rule {rule.name} crashed",
                       )]

                   if violations:
                       rules_failed += 1
                       all_violations.extend(violations)
                   else:
                       rules_passed += 1

               # Sort: CRITICAL > WARNING > SUGGESTION > INFO
               severity_order = {
                   RuleSeverity.CRITICAL: 0,
                   RuleSeverity.WARNING: 1,
                   RuleSeverity.SUGGESTION: 2,
                   RuleSeverity.INFO: 3,
               }
               all_violations.sort(key=lambda v: severity_order[v.severity])

               elapsed = (time.monotonic() - start) * 1000

               return DesignRuleReport(
                   violations=tuple(all_violations),
                   schematic_path=getattr(topology, "schematic_path", ""),
                   rules_run=rules_run,
                   rules_passed=rules_passed,
                   rules_failed=rules_failed,
                   elapsed_ms=elapsed,
               )

           def add_rule(self, rule: DesignRule) -> None:
               """Add a rule to the engine."""
               self._rules.append(rule)

           def disable_rule(self, rule_name: str) -> None:
               """Disable a rule by name."""
               self._disabled.add(rule_name)

           def enable_rule(self, rule_name: str) -> None:
               """Re-enable a previously disabled rule."""
               self._disabled.discard(rule_name)

           @property
           def rule_names(self) -> list[str]:
               """List all registered rule names."""
               return [r.name for r in self._rules]
       ```

       DOMAIN-04: 8 domain-specific design rules that go beyond
       KiCad's built-in ERC/DRC checks.

       Rules:
         BYPASS_CAP_01: Decoupling caps within 5mm of IC power pins
         FEEDBACK_01: Op-amp feedback loop has appropriate compensation cap
         IMPEDANCE_01: High-speed nets have controlled impedance
         THERMAL_01: Power components have thermal relief
         GROUND_01: Star ground topology for audio circuits
         POWER_01: Power supply filtering adequate for IC current draw
         SIGNAL_01: Input protection on externally-connected nets
         LAYOUT_01: Critical signal paths minimize via count

       Usage:
           from kicad_agent.analysis.builtin_rules import get_builtin_rules

           rules = get_builtin_rules()
           engine = DesignRuleEngine(rules=rules)
           report = engine.run(topology)
       """
       from __future__ import annotations

       import logging
       import re
       from typing import Any

       from kicad_agent.analysis.design_rules import (
           DesignRule,
           DesignRuleViolation,
           RuleCategory,
           RuleSeverity,
       )

       logger = logging.getLogger(__name__)


       # ---------------------------------------------------------------------------
       # Shared helpers
       # ---------------------------------------------------------------------------

       _IC_LIB_PATTERNS = (
           "NE5532", "TL072", "LM358", "THAT4301", "CD4066", "CD4060",
           "RP2040", "PT2399", "LM7812", "LM7912", "LM7805", "OP07",
           "OPA2134", "RC4558", "JRC4558", "BA6110", "LM13700", "LF353",
       )

       _OPAMP_LIB_PATTERNS = (
           "NE5532", "TL072", "LM358", "OP07", "OPA2134", "RC4558",
           "JRC4558", "LF353",
       )

       _POWER_NET_PREFIXES = (
           "+9V", "-9V", "+12V", "-12V", "+15V", "-15V",
           "+3V3", "+5V", "-5V", "VCC", "VDD", "VEE",
       )

       _AUDIO_NET_PATTERNS = (
           "AUDIO", "IN", "OUT", "SIGNAL", "LINE", "MIC",
           "MIX", "EQ", "FX", "SEND", "RETURN", "AUX",
       )

       _HIGH_SPEED_NET_PATTERNS = (
           "SPI", "SCLK", "MOSI", "MISO", "SS", "CS",
           "I2C", "SDA", "SCL", "UART", "TX", "RX",
           "USB", "DP", "DM", "MIDI",
       )

       _POWER_IC_PATTERNS = (
           "LM78", "LM79", "LM317", "LM337", "7805", "7812", "7912",
       )


       def _is_ic(lib_id: str) -> bool:
           return any(p.upper() in lib_id.upper() for p in _IC_LIB_PATTERNS)

       def _is_opamp(lib_id: str) -> bool:
           return any(p.upper() in lib_id.upper() for p in _OPAMP_LIB_PATTERNS)

           return "Device:R" in lib_id and "Device:R_S" not in lib_id

       def _is_audio_net(net_name: str) -> bool:
           return any(p in net_name.upper() for p in _AUDIO_NET_PATTERNS)

       def _is_high_speed_net(net_name: str) -> bool:
           return any(p in net_name.upper() for p in _HIGH_SPEED_NET_PATTERNS)

       def _parse_cap_value(value: str) -> float | None:
           """Parse capacitor value string to farads. Returns None if unparseable."""
           value = value.strip()
           multipliers = {"f": 1, "pF": 1e-12, "p": 1e-12, "nF": 1e-9, "n": 1e-9,
                          "uF": 1e-6, "u": 1e-6, "mF": 1e-3, "m": 1e-3}
           for suffix, mult in sorted(multipliers.items(), key=lambda x: -len(x[0])):
               if value.upper().endswith(suffix.upper()):
                   try:
                       return float(value[:-len(suffix)]) * mult
                   except ValueError:
                       continue
           # Try bare number (assume picofarads)
           try:
               return float(value) * 1e-12
           except ValueError:
               return None

       class BypassCapRule(DesignRule):
           """BYPASS_CAP_01: ICs must have decoupling capacitors on power pins.

           Checks that every IC has at least one ceramic capacitor (100nF typical)
           on each of its power supply nets. Audio processing ICs without bypass
           caps are flagged as CRITICAL; general ICs as WARNING.

           Config:
               max_distance_mm: Maximum distance between IC and cap (default 5.0).
                                 Distance is approximated by net connectivity
                                 (same net = within range).
               min_capacitance_pf: Minimum bypass cap value in pF (default 100).
           """

           name = "BYPASS_CAP_01"
           category = RuleCategory.BYPASS_CAPS
           default_severity = RuleSeverity.WARNING
           description = "ICs must have decoupling capacitors on power pins"

           def check(self, topology: Any, config: dict[str, Any] | None = None) -> list[DesignRuleViolation]:
               config = config or {}
               violations = []

               for comp in topology.components:
                   if not _is_ic(comp.lib_id):
                       continue
                   power_nets = [n for n in comp.pin_nets.values() if _is_power_net(n)]
                   if not power_nets:
                       continue

                   for pnet in power_nets:
                       # Find if any cap shares this power net
                       has_bypass = False
                       for other in topology.components:
                           if other.ref == comp.ref:
                               continue
                           if _is_cap(other.lib_id):
                               other_nets = set(other.pin_nets.values())
                               if pnet in other_nets:
                                   has_bypass = True
                                   break

                       if not has_bypass:
                           violations.append(DesignRuleViolation(
                               rule_id=self.name,
                               description=f"{comp.ref} ({comp.lib_id}) power pin on {pnet} "
                                          f"has no bypass capacitor",
                               severity=self.default_severity,
                               location=comp.ref,
                               suggestion=f"Add 100nF ceramic cap between {pnet} and GND near {comp.ref}",
                               affected_components=(comp.ref,),
                               details={"power_net": pnet, "ic_lib_id": comp.lib_id},
                           ))

               return violations


       class FeedbackCompRule(DesignRule):
           """FEEDBACK_01: Op-amp feedback loops should have compensation capacitors.

           Checks that op-amps with feedback networks include a small capacitor
           (typically 10-22pF) across the feedback resistor to prevent oscillation.

           Algorithm:
           1. Find op-amps
           2. Check if inverting input shares a net with output (feedback present)
           3. Check if a capacitor also shares that feedback net
           4. No cap -> SUGGESTION
           """

           name = "FEEDBACK_01"
           category = RuleCategory.FEEDBACK
           default_severity = RuleSeverity.SUGGESTION
           description = "Op-amp feedback loops should have compensation capacitors"

           def check(self, topology: Any, config: dict[str, Any] | None = None) -> list[DesignRuleViolation]:
               violations = []

               for comp in topology.components:
                   if not _is_opamp(comp.lib_id):
                       continue
                   inv_net = comp.pin_nets.get("-") or comp.pin_nets.get("INV")
                   out_net = comp.pin_nets.get("OUT")
                   if not inv_net or not out_net or inv_net != out_net:
                       # Check if inv and out share a net through another component
                       inv_nets = set(n for n in comp.pin_nets.values() if n == inv_net)
                       out_nets = set(n for n in comp.pin_nets.values() if n == out_net)
                       # Feedback = inverting input and output share components
                       if not inv_net or not out_net:
                           continue

                   # Check if feedback exists: inverting input net has a resistor
                   # connected to output net
                   has_feedback_r = False
                   has_comp_cap = False
                   for other in topology.components:
                       other_nets = set(other.pin_nets.values())
                       if inv_net in other_nets and out_net in other_nets:
                           if _is_resistor(other.lib_id):
                               has_feedback_r = True
                           elif _is_cap(other.lib_id):
                               has_comp_cap = True

                   if has_feedback_r and not has_comp_cap:
                       violations.append(DesignRuleViolation(
                           rule_id=self.name,
                           description=f"{comp.ref} ({comp.lib_id}) has feedback resistor "
                                      f"but no compensation capacitor on net {inv_net}",
                           severity=self.default_severity,
                           location=comp.ref,
                           suggestion=f"Add 10-22pF capacitor across feedback resistor "
                                      f"on {inv_net} to prevent oscillation",
                           affected_components=(comp.ref,),
                       ))

               return violations


       class ImpedanceRule(DesignRule):
           """IMPEDANCE_01: High-speed nets should have controlled impedance.

           Flags nets with high-speed protocol names (SPI, I2C, UART, USB, MIDI)
           that don't have series termination resistors.

           Severity: INFO for low-speed, WARNING for high-speed protocols.
           """

           name = "IMPEDANCE_01"
           category = RuleCategory.IMPEDANCE
           default_severity = RuleSeverity.INFO
           description = "High-speed nets should have controlled impedance"

           def check(self, topology: Any, config: dict[str, Any] | None = None) -> list[DesignRuleViolation]:
               violations = []
               for net in topology.nets:
                   if not _is_high_speed_net(net.name):
                       continue
                   # Check if any resistor is on this net (series termination)
                   has_termination = False
                   for pin_ref, pin_name in net.connected_pins:
                       for comp in topology.components:
                           if comp.ref == pin_ref and _is_resistor(comp.lib_id):
                               has_termination = True
                               break
                   if not has_termination:
                       severity = RuleSeverity.WARNING if "SPI" in net.name.upper() or "USB" in net.name.upper() else RuleSeverity.INFO
                       violations.append(DesignRuleViolation(
                           rule_id=self.name,
                           description=f"High-speed net {net.name} has no series termination resistor",
                           severity=severity,
                           location=net.name,
                           suggestion=f"Add 22-33 ohm series termination resistor on {net.name} "
                                      f"for impedance matching",
                           affected_components=tuple(pr for pr, pn in net.connected_pins),
                       ))
               return violations

       class GroundRule(DesignRule):
           """GROUND_01: Audio circuits should use star ground topology.

           Checks that audio circuits with multiple ground returns use
           a star ground point rather than daisy-chained grounds.

           Algorithm:
           1. Check if multiple ground nets exist (GND, GNDA, AGND)
           2. If yes, check if they connect at a single point
           3. Multiple separate ground nets without connection -> WARNING

           Severity: WARNING for audio circuits, INFO otherwise.
           """

           name = "GROUND_01"
           category = RuleCategory.GROUND
           default_severity = RuleSeverity.WARNING
           description = "Audio circuits should use star ground topology"

           def check(self, topology: Any, config: dict[str, Any] | None = None) -> list[DesignRuleViolation]:
               violations = []
               ground_nets = set()
               _GROUND_NAMES = ("GND", "GNDA", "AGND", "PGND", "CHASSIS", "EARTH")
               for net in topology.nets:
                   if any(g == net.name.upper() for g in _GROUND_NAMES):
                       ground_nets.add(net.name)

               if len(ground_nets) <= 1:
                   return violations  # Single ground is fine

               # Check if ground nets connect through any component
               connected_pairs = set()
               for comp in topology.components:
                   comp_nets = set(comp.pin_nets.values())
                   gnd_in_comp = comp_nets & ground_nets
                   if len(gnd_in_comp) >= 2:
                       # This component bridges two ground nets
                       connected_pairs.update(
                           tuple(sorted((a, b)))
                           for a in gnd_in_comp for b in gnd_in_comp if a != b
                       )

               # Any unconnected ground pairs?
               all_pairs = {
                   tuple(sorted((a, b)))
                   for a in ground_nets for b in ground_nets if a != b
               }
               unconnected = all_pairs - connected_pairs
               if unconnected:
                   for pair in sorted(unconnected):
                       violations.append(DesignRuleViolation(
                           rule_id=self.name,
                           description=f"Ground nets {pair[0]} and {pair[1]} are not connected "
                                      f"-- consider star ground topology",
                           severity=self.default_severity,
                           location=f"{pair[0]} / {pair[1]}",
                           suggestion=f"Connect {pair[0]} and {pair[1]} at a single star ground point "
                                      f"to avoid ground loops",
                       ))
               return violations


       class PowerFilterRule(DesignRule):
           """POWER_01: Power supply filtering should be adequate for IC current draw.

           Checks that power rails have bulk capacitors for energy storage
           and high-frequency decoupling.

           Algorithm:
           1. Find all power nets
           2. For each power net, check total capacitance
           3. If no bulk cap (>1uF) on a power net with ICs -> WARNING

           Severity: WARNING.
           """

           name = "POWER_01"
           category = RuleCategory.POWER
           default_severity = RuleSeverity.WARNING
           description = "Power supply filtering should be adequate for IC current draw"

           def check(self, topology: Any, config: dict[str, Any] | None = None) -> list[DesignRuleViolation]:
               violations = []
               power_nets = {}
               for net in topology.nets:
                   if _is_power_net(net.name):
                       power_nets[net.name] = net

               for net_name, net in power_nets.items():
                   has_bulk = False
                   has_ceramic = False
                   for pin_ref, pin_name in net.connected_pins:
                       for comp in topology.components:
                           if comp.ref == pin_ref and _is_cap(comp.lib_id):
                               cap_f = _parse_cap_value(comp.value)
                               if cap_f is not None:
                                   if cap_f >= 1e-6:
                                       has_bulk = True
                                   if cap_f <= 1e-6:
                                       has_ceramic = True

                   if not has_bulk:
                       violations.append(DesignRuleViolation(
                           rule_id=self.name,
                           description=f"Power net {net_name} has no bulk decoupling capacitor (>1uF)",
                           severity=self.default_severity,
                           location=net_name,
                           suggestion=f"Add 10uF electrolytic capacitor on {net_name} for bulk decoupling",
                           affected_components=tuple(pr for pr, pn in net.connected_pins),
                       ))
               return violations


       class InputProtectionRule(DesignRule):
           """SIGNAL_01: Input nets should have protection components.

           Flags externally-connected input nets (those with names suggesting
           external connection) that don't have series resistors, clamp diodes,
           or other protection.

           Severity: SUGGESTION.
           """

           name = "SIGNAL_01"
           category = RuleCategory.SIGNAL
           default_severity = RuleSeverity.SUGGESTION
           description = "Input nets should have protection components"

           def check(self, topology: Any, config: dict[str, Any] | None = None) -> list[DesignRuleViolation]:
               violations = []
               _INPUT_NET_PATTERNS = ("IN", "INPUT", "EXT_", "EXT", "JACK", "CONNECTOR")

               for net in topology.nets:
                   is_input = any(p in net.name.upper() for p in _INPUT_NET_PATTERNS)
                   if not is_input:
                       continue

                   has_protection = False
                   for pin_ref, pin_name in net.connected_pins:
                       for comp in topology.components:
                           if comp.ref == pin_ref:
                               if _is_resistor(comp.lib_id):
                                   has_protection = True
                               elif "Device:D" in comp.lib_id or "Device:Zener" in comp.lib_id:
                                   has_protection = True
                               elif "TVS" in comp.lib_id.upper():
                                   has_protection = True

                   if not has_protection and len(net.connected_pins) >= 2:
                       violations.append(DesignRuleViolation(
                           rule_id=self.name,
                           description=f"Input net {net_name} has no protection components "
                                      f"(no series resistor, clamp diodes, or TVS)",
                           severity=self.default_severity,
                           location=net.name,
                           suggestion=f"Consider adding series resistor or ESD protection diodes on {net.name}",
                           affected_components=tuple(pr for pr, pn in net.connected_pins),
                       ))
               return violations
